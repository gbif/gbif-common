1   package org.gbif.utils.file;
2   
3   import java.io.IOException;
4   import java.io.InputStream;
5   
6   /**
7    * A wrapper for an input stream that removes UTF8 BOM sequences at the start of the file.
8    * UTF8 BOMs can cause XML parser to fall over with a "Content is not allowed in prolog" Exception.
9    * See:
10   * <ul>
11   *  <li>http://bugs.java.com/bugdatabase/view_bug.do?bug_id=4508058</li>;
12   *  <li>https://de.wikipedia.org/wiki/Byte_Order_Mark</li>
13   * </ul>
14   *
15   * @deprecated use org.apache.commons.io.input.BOMInputStream instead
16   */
17  @Deprecated
18  public class BomSafeInputStreamWrapper extends InputStream {
19  
20    private static final int BUFFER_SIZE = 4;
21    private final InputStream stream;
22    private final byte[] buffer = new byte[BUFFER_SIZE];
23    private int pointer = 0;
24  
25    public BomSafeInputStreamWrapper(InputStream stream) {
26      this.stream = stream;
27      skipBom();
28    }
29  
30    @Override
31    public int read() throws IOException {
32      if (pointer < BUFFER_SIZE) {
33        pointer++;
34        return buffer[pointer - 1];
35      } else {
36        return stream.read();
37      }
38    }
39  
40    private void skipBom() {
41      try {
42        stream.read(buffer, 0, BUFFER_SIZE);
43        if (CharsetDetection.hasUTF16BEBom(buffer) || CharsetDetection.hasUTF16LEBom(buffer)) {
44          // SQX Parser handles UTF16 BOMs fine
45          pointer = 2;
46        } else if (CharsetDetection.hasUTF8Bom(buffer)) {
47          pointer = 3;
48        }
49      } catch (IOException ignored) {
50      }
51    }
52  
53  }
