74    /**
75     * Tries to decompress a file into a newly created temporary directory, trying gzip or zip regardless of the filename
76     * or its suffix.
77     *
78     * @return folder containing all decompressed files
79     */
80    public static File decompressFile(File compressedFile) throws IOException, UnsupportedCompressionType {
81      // create empty tmp dir
82      File dir = File.createTempFile("gbif-", null);
83      if (dir.exists() && !dir.delete()) {
84        throw new IOException("Couldn't delete temporary directory");
85      }
86  
87      if (!dir.mkdirs()) {
88        throw new IOException("Couldn't create temporary directory for decompression");
89      }
90  
91      // decompress
92      decompressFile(dir, compressedFile);
93  
94      return dir;
95    }

107   /**
108    * Tries to decompress a file trying gzip or zip regardless of the filename or its suffix.
109    *
110    * @param directory      directory where archive's contents will be decompressed to
111    * @param compressedFile compressed file
112    *
113    * @return list of files that have been extracted or null an empty list if archive couldn't be decompressed
114    *
115    * @throws IOException                if problem occurred reading compressed file, or directory couldn't be written
116    *                                    to
117    * @throws UnsupportedCompressionType if the compression type wasn't recognized
118    */
119   public static List<File> decompressFile(File directory, File compressedFile, boolean keepSubdirectories)
120     throws IOException, UnsupportedCompressionType {
121     List<File> files = null;
122 
123     // Test before trying gzip format
124     if (isGzipFormat(compressedFile)) {
125       try {
126         files = ungzipFile(directory, compressedFile);
127       } catch (Exception e) {
128         LOG.debug("No gzip compression");
129       }
130     }
131 
132     // Then try zip
133     if (files == null) {
134       try {
135         files = unzipFile(directory, compressedFile, keepSubdirectories);
136       } catch (ZipException e) {
137         LOG.debug("No zip compression");
138         throw new UnsupportedCompressionType("Unknown compression type. Neither zip nor gzip", e);
139       }
140     }
141 
142     return files;
143   }

145   /**
146    * Check the file's first two bytes, to see if they are the gzip magic number.
147    * @param compressedFile compressed file
148    * @return               true if the file is in gzip format
149    * @throws IOException   if a problem occurred reading compressed file
150    */
151   private static boolean isGzipFormat(File compressedFile) throws IOException {
152     try (RandomAccessFile file = new RandomAccessFile(compressedFile, "r")) {
153       return GZIPInputStream.GZIP_MAGIC == (file.read() & 0xff | ((file.read() << 8) & 0xff00));
154     }
155   }

157   /**
158    * Extracts a gzipped file. Subdirectories or hidden files (i.e. files starting with a dot) are being ignored.
159    *
160    * @param directory where the file should be extracted to
161    * @param zipFile   to extract
162    *
163    * @return a list of all created files
164    */
165   public static List<File> ungzipFile(File directory, File zipFile) throws IOException {
166     List<File> files = new ArrayList<File>();
167     TarArchiveInputStream in = new TarArchiveInputStream(new GZIPInputStream(new FileInputStream(zipFile)));
168     try {
169       TarArchiveEntry entry = in.getNextTarEntry();
170       while (entry != null) {
171         if (entry.isDirectory()) {
172           LOG.debug("TAR archive contains directories which are being ignored");
173           entry = in.getNextTarEntry();
174           continue;
175         }
176         String fn = new File(entry.getName()).getName();
177         if (fn.startsWith(".")) {
178           LOG.debug("TAR archive contains a hidden file which is being ignored");
179           entry = in.getNextTarEntry();
180           continue;
181         }
182         File targetFile = new File(directory, fn);
183         if (targetFile.exists()) {
184           LOG.warn("TAR archive contains duplicate filenames, only the first is being extracted");
185           entry = in.getNextTarEntry();
186           continue;
187         }
188         LOG.debug("Extracting file: {} to: {}", entry.getName(), targetFile.getAbsolutePath());
189         FileOutputStream out = new FileOutputStream(targetFile);
190         try {
191           IOUtils.copy(in, out);
192           out.close();
193         } finally {
194           IOUtils.closeQuietly(out);
195         }
196         files.add(targetFile);
197       }
198     } finally {
199       in.close();
200     }
201     return files;
202   }

204   /**
205    * Gunzip a file.  Use this method with isTarred false if the gzip contains a single file.  If it's a gzip
206    * of a tar archive pass true to isTarred (or call @ungzipFile(directory, zipFile) which is what this method
207    * just redirects to for isTarred).
208    *
209    * @param directory the output directory for the uncompressed file(s)
210    * @param zipFile   the gzip file
211    * @param isTarred  true if the gzip contains a tar archive
212    *
213    * @return a List of the uncompressed file name(s)
214    *
215    * @throws IOException if reading or writing fails
216    */
217   public static List<File> ungzipFile(File directory, File zipFile, boolean isTarred) throws IOException {
218     if (isTarred) return ungzipFile(directory, zipFile);
219 
220     List<File> files = new ArrayList<File>();
221     GZIPInputStream in = null;
222     BufferedOutputStream dest = null;
223     try {
224       in = new GZIPInputStream(new FileInputStream(zipFile));
225 
226       // assume that the gzip filename is the filename + .gz
227       String unzippedName = zipFile.getName().substring(0, zipFile.getName().lastIndexOf("."));
228       File outputFile = new File(directory, unzippedName);
229       LOG.debug("Extracting file: {} to: {}", unzippedName, outputFile.getAbsolutePath());
230       FileOutputStream fos = new FileOutputStream(outputFile);
231 
232       dest = new BufferedOutputStream(fos, BUFFER);
233       int count;
234       byte[] data = new byte[BUFFER];
235       while ((count = in.read(data, 0, BUFFER)) != -1) {
236         dest.write(data, 0, count);
237       }
238       files.add(outputFile);
239     } finally {
240       if (in != null) in.close();
241       if (dest != null) {
242         dest.flush();
243         dest.close();
244       }
245     }
246 
247     return files;
248   }

288   /**
289    * Creates a zip archive from a given collection of files.
290    * In order to preserve paths in the archive a rootContext can be specified which will be removed from the individual
291    * zip entries. For example a rootContext of /home/freak with a file /home/freak/photo/birthday.jpg to be zipped
292    * will result in a zip entry with a path photo/birthday.jpg.
293    *
294    * @param files to be included in the zip archive
295    * @param rootContext optional path to be removed from each file
296    * @param zipFile the zip file to be created
297    * @throws IOException
298    */
299   public static void zipFiles(Collection<File> files, File rootContext, File zipFile) throws IOException {
300     if (files.isEmpty()) {
301       LOG.info("no files to zip.");
302     } else {
303       try {
304         BufferedInputStream origin = null;
305         FileOutputStream dest = new FileOutputStream(zipFile);
306         ZipOutputStream out = new ZipOutputStream(new BufferedOutputStream(dest));
307         // out.setMethod(ZipOutputStream.DEFLATED);
308         byte[] data = new byte[BUFFER];
309         for (File f : files) {
310           LOG.debug("Adding file {} to archive", f);
311           FileInputStream fi = new FileInputStream(f);
312           origin = new BufferedInputStream(fi, BUFFER);
313 
314           String zipPath = StringUtils.removeStart(f.getAbsolutePath(), rootContext.getAbsolutePath() + File.separator);
315           ZipEntry entry = new ZipEntry(zipPath);
316           out.putNextEntry(entry);
317           int count;
318           while ((count = origin.read(data, 0, BUFFER)) != -1) {
319             out.write(data, 0, count);
320           }
321           origin.close();
322         }
323         out.finish();
324         out.close();
325       } catch (IOException e) {
326         LOG.error("IOException while zipping files: {}", files);
327         throw e;
328       }
329     }
330   }

332   /**
333    * Extracts a zipped file into a target directory. If the file is wrapped in a root directory, this is removed by
334    * default. Other subdirectories are ignored according to the parameter keepSubdirectories.
335    * </br>
336    * The following types of files are also ignored by default:
337    * i) hidden files (i.e. files starting with a dot)
338    * ii) Apple resource fork (__MACOSX), including its subdirectories and subfiles
339    *
340    * @param directory          where the zipped file and its subdirectories should be extracted to
341    * @param zipFile            to extract
342    * @param keepSubdirectories whether to preserve subdirectories or not
343    *
344    * @return a list of all created files and directories extracted to target directory
345    */
346   public static List<File> unzipFile(File directory, File zipFile, boolean keepSubdirectories) throws IOException {
347     LOG.debug("Unzipping archive " + zipFile.getName() + " into directory: " + directory.getAbsolutePath());
348 
349     // This is changed from using ZipFile to a ZipInputStream since Java 8u192 can't open certain Zip64 files.
350     // https://bugs.openjdk.java.net/browse/JDK-8186464
351     try (FileInputStream fInput = new FileInputStream(zipFile);
352          ZipInputStream zipInput = new ZipInputStream(fInput)) {
353       ZipEntry entry;
354 
355       while ((entry = zipInput.getNextEntry()) != null) {
356         // ignore resource fork directories and subfiles
357         if (entry.getName().toUpperCase().contains(APPLE_RESOURCE_FORK)) {
358           LOG.debug("Ignoring resource fork file: " + entry.getName());
359         }
360         // ignore directories and hidden directories (e.g. .svn) (based on flag)
361         else if (entry.isDirectory()) {
362           if (isHiddenFile(new File(entry.getName()))) {
363             LOG.debug("Ignoring hidden directory: " + entry.getName());
364           } else if (keepSubdirectories) {
365             new File(directory, entry.getName()).mkdir();
366           } else {
367             LOG.debug("Ignoring (sub)directory: " + entry.getName());
368           }
369         }
370         // ignore hidden files
371         else {
372           if (isHiddenFile(new File(entry.getName()))) {
373             LOG.debug("Ignoring hidden file: " + entry.getName());
374           } else {
375             File targetFile = (keepSubdirectories) ? new File(directory, entry.getName())
376               : new File(directory, new File(entry.getName()).getName());
377             // ensure parent folder always exists, and extract file
378             createParentFolder(targetFile);
379 
380             LOG.debug("Extracting file: {} to: {}", entry.getName(), targetFile.getAbsolutePath());
381             try (OutputStream out = new BufferedOutputStream(new FileOutputStream(targetFile))) {
382               IOUtils.copy(zipInput, out);
383             }
384           }
385         }
386       }
387     }
388     // remove the wrapping root directory and flatten structure
389     if (keepSubdirectories) {
390       removeRootDirectory(directory);
391     }
392     return (directory.listFiles() == null) ? new ArrayList<File>() : Arrays.asList(directory.listFiles());
393   }

395   /**
396    * @return true if file is a hidden file or directory, or if any of its parent directories are hidden checking
397    * recursively
398    */
399   private static boolean isHiddenFile(File f) {
400     if (f.getName().startsWith(".")) {
401       return true;
402     } else if (f.getParentFile() != null) {
403       return isHiddenFile(f.getParentFile());
404     }
405     return false;
406   }

408   /**
409    * Removes a wrapping root directory and flatten its structure by moving all that root directory's files and
410    * subdirectories up to the same level as the root directory.
411    */
412   private static void removeRootDirectory(File directory) {
413     File[] rootFiles = directory.listFiles((FileFilter) HiddenFileFilter.VISIBLE);
414     if (rootFiles.length == 1) {
415       File root = rootFiles[0];
416       if (root.isDirectory()) {
417         LOG.debug("Removing single root folder {} found in decompressed archive", root.getAbsoluteFile());
418         for (File f : org.apache.commons.io.FileUtils.listFilesAndDirs(root, TrueFileFilter.TRUE, TrueFileFilter.TRUE)) {
419           File f2 = new File(directory, f.getName());
420           f.renameTo(f2);
421         }
422         root.delete();
423       }
424     }
425   }

427   /**
428    * Make parent folder.
429    *
430    * @param file destination file
431    */
432   private static void createParentFolder(File file) {
433     File parent = new File(file.getParent());
434     if (!parent.exists()) {
435       LOG.debug((parent.mkdirs()) ? "Created parent directory: " + parent.getAbsolutePath()
436         : "Failed to create parent directory: " + parent.getAbsolutePath());
437     }
