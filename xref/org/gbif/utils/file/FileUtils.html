93    /**
94     * Reads a file and returns a unique set of multiple columns from lines which are no comments (starting with #) and
95     * trims whitespace.
96     *
97     * @param source the UTF-8 encoded text file with tab delimited columns
98     * @param resultSet the set implementation to be used. Will not be cleared before reading!
99     * @param column variable length argument of column indices to process
100    * @return set of column rows
101    */
102   public static Set<String> columnsToSet(InputStream source, Set<String> resultSet, int... column) throws IOException {
103     LineIterator lines = getLineIterator(source);
104     int maxCols = 0;
105     for (int c : column) {
106       if (c > maxCols) {
107         maxCols = c;
108       }
109     }
110     while (lines.hasNext()) {
111       String line = lines.nextLine().trim();
112       // ignore comments
113       if (!ignore(line)) {
114         String[] parts = TAB_DELIMITED.split(line);
115         if (maxCols <= parts.length) {
116           for (int c : column) {
117             String cell = parts[c].trim();
118             resultSet.add(cell);
119           }
120         }
121       }
122     }
123     return resultSet;
124   }

126   public static void copyStreams(InputStream in, OutputStream out) throws IOException {
127     // write the file to the file specified
128     int bytesRead;
129     byte[] buffer = new byte[8192];
130 
131     while ((bytesRead = in.read(buffer, 0, 8192)) != -1) {
132       out.write(buffer, 0, bytesRead);
133     }
134 
135     out.close();
136     in.close();
137   }

241   /**
242    * Converts the byte size into human-readable format.
243    * Support both SI and byte format.
244    */
245   public static String humanReadableByteCount(long bytes, boolean si) {
246     int unit = si ? 1000 : 1024;
247     if (bytes < unit) {
248       return bytes + " B";
249     }
250     int exp = (int) (Math.log(bytes) / Math.log(unit));
251     String pre = (si ? "kMGTPE" : "KMGTPE").charAt(exp - 1) + (si ? "" : "i");
252     return String.format("%.1f %sB", bytes / Math.pow(unit, exp), pre);
253   }

276     BufferedInputStream f = new BufferedInputStream(new FileInputStream(file), bufferSize);
277 
278     int b;

282       }
283       bbuf.put((byte) b);
284     }
285     f.close();
286

365    * Read a hashmap from a tab delimited utf8 input stream using the row number as an integer value and the entire row

373     while (lines.hasNext()) {
374       row++;

377       if (!ignore(line)) {
378         result.put(line, row.toString());
379       }
380     }

384   /**
385    * Read a hashmap from a tab delimited utf8 file, ignoring commented rows starting with #.
386    *
387    * @param source tab delimited input stream to read
388    * @param key column number to use as key
389    * @param value column number to use as value
390    * @param trimToNull if true trims map entries to null
391    */
392   public static Map<String, String> streamToMap(InputStream source, Map<String, String> result, int key, int value,
393     boolean trimToNull) throws IOException {
394     LineIterator lines = getLineIterator(source);
395     int maxCols = key > value ? key : value + 1;
396     while (lines.hasNext()) {
397       String line = lines.nextLine();
398       // ignore comments
399       if (!ignore(line)) {
400         String[] parts = TAB_DELIMITED.split(line);
401         if (maxCols <= parts.length) {
402           if (trimToNull) {
403             result.put(StringUtils.trimToNull(parts[key]), StringUtils.trimToNull(parts[value]));
404           } else {
405             result.put(parts[key], parts[value]);
406           }
407         }
408       }
409     }
410     return result;
411   }

453   /**
454    * For the given list, finds the index of the lowest value using the given comparator.
455    *
456    * @param values To compare
457    * @param comparator To use
458    * @return The index of the lowest value, or -1 if they are all null
459    */
460   static int lowestValueIndex(List<String> values, Comparator<String> comparator) {
461     int index = 0;
462     String lowestValue = null;
463     for (int i = 0; i < values.size(); i++) {
464       String value = values.get(i);
465       if (lowestValue == null) {
466         lowestValue = value;
467         index = i;
468       } else if (comparator.compare(lowestValue, value) > 0) {
469         lowestValue = value;
470         index = i;
471       }
472     }
473 
474     return lowestValue == null ? -1 : index;
475   }

498   /**
499    * Merges a list of intermediary sort chunk files into a single sorted file. On completion, the intermediary sort
500    * chunk files are deleted.
501    *
502    * @param sortFiles sort chunk files to merge
503    * @param sortedFileWriter writer to merge to. Can already be open and contain data
504    * @param lineComparator To use when determining the order (reuse the one that was used to sort the individual
505    *        files)
506    */
507   public void mergedSortedFiles(List<File> sortFiles, FileWriter sortedFileWriter, Comparator<String> lineComparator)
508     throws IOException {
509     List<BufferedReader> partReaders = new LinkedList<BufferedReader>();
510     try {
511       List<String> partReaderLine = new LinkedList<String>();
512       for (File f : sortFiles) {
513         partReaders.add(new BufferedReader(new FileReader(f)));
514       }
515       boolean moreData = false;
516       // load first line in
517       for (BufferedReader partReader : partReaders) {
518         String partLine = partReader.readLine();
519         if (partLine != null) {
520           moreData = true;
521         }
522         // we still add the "null" to keep the partReaders and partLineReader indexes in sync - ALWAYS
523         partReaderLine.add(partLine);
524       }
525       // keep going until all readers are exhausted
526       while (moreData) {
527         int index = lowestValueIndex(partReaderLine, lineComparator);
528         if (index >= 0) {
529           sortedFileWriter.write(partReaderLine.get(index));
530           sortedFileWriter.write("\n");
531           BufferedReader r = partReaders.get(index);
532           String partLine = r.readLine();
533           // TODO: Synchronization on local variable?
534           synchronized (partReaderLine) {
535             partReaderLine.add(index, partLine);
536             partReaderLine.remove(index + 1);
537           }
538         } else {
539           moreData = false;
540         }
541       }
542     } finally {
543       for (BufferedReader b : partReaders) {
544         try {
545           b.close();
546         } catch (RuntimeException e) {
547         }
548       }
549       // I assume it periodically flushes anyway, so only need to do once at end...
550       sortedFileWriter.flush();
551       sortedFileWriter.close();
552       // delete (intermediary) sort chunk files, only the sorted file remains
553       for (File f : sortFiles) {
554         f.delete();
555       }
556     }
557   }

559   /**
560    * Sorts the input file into the output file using the supplied delimited line parameters.
561    *
562    * This method is not reliable when the sort field may contain Unicode codepoints outside the Basic Multilingual Plane,
563    * i.e. above \uFFFF. In that case, the sort order differs from Java's String sort order.  This should not be a problem
564    * for most usage; the Supplementary Multilingual Planes contain ancient scripts, emojis, arrows and so on.
565    *
566    * @param input To sort
567    * @param sorted The sorted version of the input excluding ignored header lines (see ignoreHeaderLines)
568    * @param column the column that keeps the values to sort on
569    * @param columnDelimiter the delimiter that separates columns in a row
570    * @param enclosedBy optional column enclosing character, e.g. a double quote for CSVs
571    * @param newlineDelimiter the chars used for new lines, usually \n, \n\r or \r
572    * @param ignoreHeaderLines number of beginning lines to ignore, e.g. headers
573    */
574   public void sort(File input, File sorted, String encoding, int column, String columnDelimiter, Character enclosedBy,
575     String newlineDelimiter, int ignoreHeaderLines) throws IOException {
576     Comparator<String> lineComparator;
577     if (enclosedBy == null) {
578       lineComparator = new LineComparator(column, columnDelimiter);
579     } else {
580       lineComparator = new LineComparator(column, columnDelimiter, enclosedBy);
581     }
582     sort(input, sorted, encoding, column, columnDelimiter, enclosedBy, newlineDelimiter, ignoreHeaderLines,
583       lineComparator, false);
584   }

586   /**
587    * Sorts the input file into the output file using the supplied delimited line parameters.
588    *
589    * This method is not reliable when the sort field may contain Unicode codepoints outside the Basic Multilingual Plane,
590    * i.e. above \uFFFF. In that case, the sort order differs from Java's String sort order.  This should not be a problem
591    * for most usage; the Supplementary Multilingual Planes contain ancient scripts, emojis, arrows and so on.
592    *
593    * TODO: This method is globally synchronized, in case multiple sorts are attempted to the same file simultaneously.
594    * This could be improved to allow synchronizing against the destination file, rather than for all sorts.
595    *
596    * @param input To sort
597    * @param sorted The sorted version of the input excluding ignored header lines (see ignoreHeaderLines)
598    * @param column the column that keeps the values to sort on
599    * @param columnDelimiter the delimiter that separates columns in a row
600    * @param enclosedBy optional column enclosing character, e.g. a double quote for CSVs
601    * @param newlineDelimiter the chars used for new lines, usually \n, \r\n or \r
602    * @param ignoreHeaderLines number of beginning lines to ignore, e.g. headers
603    * @param lineComparator used to sort the output
604    * @param ignoreCase ignore case order, this parameter couldn't have any effect if the LineComparator is used
605    */
606   public void sort(File input, File sorted, String encoding, int column, String columnDelimiter, Character enclosedBy,
607     String newlineDelimiter, int ignoreHeaderLines, Comparator<String> lineComparator, boolean ignoreCase)
608     throws IOException {
609     LOG.debug("Sorting " + input.getAbsolutePath() + " as new file " + sorted.getAbsolutePath());
610     if (encoding == null) {
611       LOG.warn("No encoding specified, assume UTF-8");
612       encoding = FileUtils.UTF8;
613     }
614     synchronized (sortLock) {
615       if (sorted.exists()) {
616         // Delete a file, which will allow processes with it open to continue reading it.
617         // The GNU sort truncates and appends, which would mean a partial read otherwise.
618         LOG.warn("Deleting existed sorted file {}", sorted.getAbsoluteFile());
619         sorted.delete();
620       }
621       // if the id is in the first column, first try sorting via shell as its the fastest we can get
622       if (!sortInGnu(input, sorted, encoding, ignoreHeaderLines, column, columnDelimiter, newlineDelimiter, ignoreCase)) {
623         LOG.debug("No GNU sort available, using native Java sorting");
624         sortInJava(input, sorted, encoding, lineComparator, ignoreHeaderLines);
625       }
626     }
627   }

629   /**
630    * Sorts the input file into the output file using the supplied lineComparator.
631    *
632    * @param input To sort
633    * @param sorted The sorted version of the input excluding ignored header lines (see ignoreHeaderLines)
634    * @param lineComparator To use during comparison
635    * @param ignoreHeaderLines number of beginning lines to ignore, e.g. headers
636    */
637   public void sortInJava(File input, File sorted, String encoding, Comparator<String> lineComparator,
638     int ignoreHeaderLines) throws IOException {
639     LOG.debug("Sorting File[" + input.getAbsolutePath() + ']');
640     long start = System.currentTimeMillis();
641     List<File> sortFiles = new LinkedList<File>();
642     BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(input), encoding));
643     List<String> headerLines = new LinkedList<String>();
644     try {
645       String line = br.readLine();
646       int fileCount = 0;
647 
648       List<String> linesToSort = new LinkedList<String>();
649       while (line != null) {
650         if (ignoreHeaderLines > 0) {
651           headerLines.add(line);
652           ignoreHeaderLines--;
653         } else {
654           linesToSort.add(line);
655 
656           // if buffer is full, then sort and write to file
657           if (linesToSort.size() == linesPerMemorySort) {
658             sortFiles.add(sortAndWrite(input, encoding, lineComparator, fileCount, linesToSort));
659             linesToSort = new LinkedList<String>();
660             fileCount++;
661           }
662         }
663         line = br.readLine();
664       }
665       // catch the last lot
666       if (!linesToSort.isEmpty()) {
667         sortFiles.add(sortAndWrite(input, encoding, lineComparator, fileCount, linesToSort));
668       }
669     } finally {
670       br.close();
671     }
672     LOG.debug(
673       sortFiles.size() + " sorted file chunks created in " + (System.currentTimeMillis() - start) / 1000 + " secs");
674 
675     // now merge the sorted files into one single sorted file
676     FileWriter sortedFileWriter = new FileWriter(sorted);
677     // first write the old header lines if existing
678     for (String h : headerLines) {
679       sortedFileWriter.write(h);
680       sortedFileWriter.write("\n");
681     }
682     mergedSortedFiles(sortFiles, sortedFileWriter, lineComparator);
683 
684     LOG.debug(
685       "File " + input.getAbsolutePath() + " sorted successfully using " + sortFiles.size() + " parts to do sorting in "
686         + (System.currentTimeMillis() - start) / 1000 + " secs");
687   }

690   /**
691    * Splits the supplied file into files of set line size and with a suffix.
692    *
693    * @param input To split up
694    * @param linesPerOutput Lines per split file
695    * @param extension The file extension to use - e.g. ".txt"
696    * @return The split files
697    */
698   public List<File> split(File input, int linesPerOutput, String extension) throws IOException {
699     LOG.debug("Splitting File[" + input.getAbsolutePath() + ']');
700     long timer = System.currentTimeMillis();
701     List<File> splitFiles = new LinkedList<File>();
702     BufferedReader br = new BufferedReader(new FileReader(input));
703     String line = br.readLine();
704     int fileCount = 0;
705     File splitFile = getChunkFile(input, fileCount);
706     fileCount++;
707     splitFiles.add(splitFile);
708     FileWriter fw = new FileWriter(splitFile);
709     try {
710       int lineCount = 0;
711       while (line != null) {
712         if (lineCount == linesPerOutput) {
713           fw.flush();
714           fw.close();
715           splitFile = getChunkFile(input, fileCount);
716           splitFiles.add(splitFile);
717           // is ok to reuse, as last one is closed, and this will always get closed - see finally below
718           fw = new FileWriter(splitFile);
719           fileCount++;
720           lineCount = 0;
721         }
722         fw.write(line);
723         fw.write("\n");
724         line = br.readLine();
725         lineCount++;
726       }
727       fw.flush();
728     } finally {
729       fw.close();
730     }
731     LOG.debug("File[" + input.getAbsolutePath() + "] split successfully into[" + splitFiles.size() + "] parts in secs["
732       + (1 + System.currentTimeMillis() - timer) / 1000 + "]");
733     return splitFiles;
734   }

736   /**
737    * Test whether we have a new enough version of GNU Sort that supports (primarily) the -k option with a start and end
738    * column.
739    *
740    * Mac OS only includes an old version of GNU sort, and will fail this test.
741    */
742   private boolean gnuSortAvailable() {
743     if (gnuSortAvailable != null) {
744       return gnuSortAvailable;
745     }
746 
747     try {
748       String command = "sort -k1,1 -t',' --ignore-case /dev/null";
749       LOG.debug("Testing capability of GNU sort with command: {}", command);
750 
751       Process process = new ProcessBuilder("/bin/sh", "-c", command).start();
752       int exitValue = process.waitFor();
753 
754       if (exitValue == 0) {
755         LOG.debug("GNU sort is capable");
756         gnuSortAvailable = true;
757       } else {
758         LOG.warn("GNU sort does not exist or is too old, and will not be used.  Sorting large files will be slow.",
759             new InputStreamUtils().readEntireStream(process.getErrorStream()).replace('\n', ' '));
760         gnuSortAvailable = false;
761       }
762     } catch (Exception e) {
763       LOG.warn("GNU sort does not exist or is too old, and will not be used.  Sorting large files will be slow.", e);
764       gnuSortAvailable = false;
765     }
766 
767     return gnuSortAvailable;
768   }

770   /**
771    * sort a text file via an external GNU sort command:
772    * sorting tabs at 3rd column, numerical reverse order
773    * sort -t$'\t' -k3 -o sorted.txt col2007.txt
774    * <p/>
775    * The GNU sort based sorting is extremely efficient and much, much faster than the current sortInJava method. It is
776    * locale aware though and we only want the native C sorting locale. See
777    * http://www.gnu.org/software/coreutils/faq/coreutils-faq.html#Sort-does-not-sort-in-normal-order_0021
778    * <p/>
779    * Example C sort order:
780    * <p/>
781    * <pre>
782    * 1 oOdontoceti
783    * 10 gGlobicephala melaena melaena Traill
784    * 100 gGlobicephala melaena melaena Traill
785    * 101 gGlobicephala melaena melaena Traill
786    * 11 pPontoporia Gray
787    * 12 pPontoporia blainvillei Gervais and d'Orbigny
788    * 120 iInia d'Orbigny
789    * 121 iInia geoffrensis Blainville
790    * 2 sSusuidae
791    * 20 cCetacea
792    * Amphiptera
793    * Amphiptera pacifica Giglioli
794    * Anarnak Lacépède
795    * Balaena mangidach Chamisso
796    * amphiptera
797    * amphiptera pacifica Giglioli
798    * anarnak Lacépède
799    * balaena mangidach Chamisso
800    * </pre>
801    */
802   protected boolean sortInGnu(File input, File sorted, String encoding, int ignoreHeaderLines, int column,
803     String columnDelimiter, String lineDelimiter, boolean ignoreCase) throws IOException {
804     String command;
805     // GNU sort is checked for use when:
806     // • line delimiter is \n
807     // • column delimiter is set and we're not using the first column
808     // • sort version is sufficient to include start and end column (-k 1,1).
809     // Use the --debug option to sort if working on this code.
810     if (lineDelimiter == null || !lineDelimiter.contains("\n") || (columnDelimiter != null && column > 0) ||
811         !gnuSortAvailable()) {
812       LOG.debug("Cannot use GNU sort on this file");
813       return false;
814     }
815 
816     // keep header rows
817     boolean success = false;
818     try {
819       LinkedList<String> cmds = new LinkedList<String>();
820       cmds.add("/bin/sh");
821       cmds.add("-c");
822       cmds.add("");
823       ProcessBuilder pb = new ProcessBuilder(cmds);
824       Map<String, String> env = pb.environment();
825       
826       //clear the environment, but keep specified temp working directory 
827       env.keySet().removeIf(key -> !(key.equals("TMPDIR")));
828       if (System.getProperty("java.io.tmpdir") != null) {
829         env.put("TMPDIR", System.getProperty("java.io.tmpdir"));
830       }
831       // make sure we use the C locale for sorting
832       env.put("LC_ALL", "C");
833 
834       String sortArgs = String.format(" %s -k%d,%d -t'%s'",
835         ignoreCase ? "--ignore-case" : "", column+1, column+1, columnDelimiter);
836 
837       if (ignoreHeaderLines > 0) {
838         // copy header lines
839         command = "head -n " + ignoreHeaderLines + ' ' + input.getAbsolutePath() + " > " + sorted.getAbsolutePath();
840         LOG.debug("Issue external command: {}", command);
841         cmds.removeLast();
842         cmds.add(command);
843         Process process = pb.start();
844         int exitValue = process.waitFor();
845         if (exitValue != 0) {
846           LOG.warn("Error sorting file (copying header lines) with GNU head");
847           return false;
848         }
849 
850         // do the sorting ignoring the header rows
851         command = "sed " + ignoreHeaderLines + "d " + input.getAbsolutePath() + " | "
852             + "sort " + sortArgs
853             + " >> " + sorted.getAbsolutePath();
854       } else {
855         // do sorting directly, we don't have header rows
856         command = "sort " + sortArgs + " -o " + sorted.getAbsolutePath() + ' ' + input.getAbsolutePath();
857       }
858 
859       LOG.debug("Issue external command: {}", command);
860       cmds.removeLast();
861       cmds.add(command);
862       Process process = pb.start();
863       // get the stdout and stderr from the command that was run
864       InputStream err = process.getErrorStream();
865       int exitValue = process.waitFor();
866       if (exitValue == 0) {
867         LOG.debug("Successfully sorted file with GNU sort");
868         success = true;
869       } else {
870         LOG.warn("Error sorting file with GNU sort");
871         InputStreamUtils isu = new InputStreamUtils();
872         System.err.append(isu.readEntireStream(err));
873       }
874     } catch (Exception e) {
875       LOG.warn("Caught Exception using GNU sort", e);
876     }
877     return success;
878   }

880   /**
881    * Sorts the lines and writes to file using the
882    *
883    * @param input File to base the name on
884    * @param lineComparator To compare the lines for sorting
885    * @param fileCount Used for the file name
886    * @param linesToSort To actually sort
887    * @return The written file
888    */
889   private File sortAndWrite(File input, String encoding, Comparator<String> lineComparator, int fileCount,
890     List<String> linesToSort) throws IOException {
891     long start = System.currentTimeMillis();
892     Collections.sort(linesToSort, lineComparator);
893     // When implementing a comparator, make it SUPER quick!!!
894     LOG.debug(
895       "Collections.sort took msec[" + (System.currentTimeMillis() - start) + "] to sort records[" + linesToSort.size()
896         + ']');
897     File sortFile = getChunkFile(input, fileCount);
898     Writer fw = new OutputStreamWriter(new FileOutputStream(sortFile), encoding);
899     try {
900       for (String s : linesToSort) {
901         fw.write(s);
902         fw.write("\n");
903       }
904     } finally {
905       fw.close();
906     }
907     return sortFile;
908   }
