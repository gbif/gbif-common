91    /**
92     * Reads a file and returns a unique set of multiple columns from lines which are no comments (starting with #) and
93     * trims whitespace.
94     *
95     * @param source the UTF-8 encoded text file with tab delimited columns
96     * @param resultSet the set implementation to be used. Will not be cleared before reading!
97     * @param column variable length argument of column indices to process
98     * @return set of column rows
99     */
100   public static Set<String> columnsToSet(InputStream source, Set<String> resultSet, int... column) throws IOException {
101     LineIterator lines = getLineIterator(source);
102     int maxCols = 0;
103     for (int c : column) {
104       if (c > maxCols) {
105         maxCols = c;
106       }
107     }
108     while (lines.hasNext()) {
109       String line = lines.nextLine().trim();
110       // ignore comments
111       if (!ignore(line)) {
112         String[] parts = TAB_DELIMITED.split(line);
113         if (maxCols <= parts.length) {
114           for (int c : column) {
115             String cell = parts[c].trim();
116             resultSet.add(cell);
117           }
118         }
119       }
120     }
121     return resultSet;
122   }

124   public static void copyStreams(InputStream in, OutputStream out) throws IOException {
125     // write the file to the file specified
126     int bytesRead;
127     byte[] buffer = new byte[8192];
128 
129     while ((bytesRead = in.read(buffer, 0, 8192)) != -1) {
130       out.write(buffer, 0, bytesRead);
131     }
132 
133     out.close();
134     in.close();
135   }

239   /**
240    * Converts the byte size into human-readable format.
241    * Support both SI and byte format.
242    */
243   public static String humanReadableByteCount(long bytes, boolean si) {
244     int unit = si ? 1000 : 1024;
245     if (bytes < unit) {
246       return bytes + " B";
247     }
248     int exp = (int) (Math.log(bytes) / Math.log(unit));
249     String pre = (si ? "kMGTPE" : "KMGTPE").charAt(exp - 1) + (si ? "" : "i");
250     return String.format("%.1f %sB", bytes / Math.pow(unit, exp), pre);
251   }

274     BufferedInputStream f = new BufferedInputStream(new FileInputStream(file), bufferSize);
275 
276     int b;

362   /**
363    * Read a hashmap from a tab delimited utf8 input stream using the row number as an integer value and the entire row
364    * as the value. Ignores commented rows starting with #.
365    *
366    * @param source tab delimited text file to read
367    */
368   public static Map<String, String> streamToMap(InputStream source, Map<String, String> result) throws IOException {
369     LineIterator lines = getLineIterator(source);
370     Integer row = 0;
371     while (lines.hasNext()) {
372       row++;
373       String line = lines.nextLine().trim();
374       // ignore comments
375       if (!ignore(line)) {
376         result.put(line, row.toString());
377       }
378     }
379     return result;

382   /**
383    * Read a hashmap from a tab delimited utf8 file, ignoring commented rows starting with #.
384    *
385    * @param source tab delimited input stream to read
386    * @param key column number to use as key
387    * @param value column number to use as value
388    * @param trimToNull if true trims map entries to null
389    */
390   public static Map<String, String> streamToMap(InputStream source, Map<String, String> result, int key, int value,
391     boolean trimToNull) throws IOException {
392     LineIterator lines = getLineIterator(source);
393     int maxCols = key > value ? key : value + 1;
394     while (lines.hasNext()) {
395       String line = lines.nextLine();
396       // ignore comments
397       if (!ignore(line)) {
398         String[] parts = TAB_DELIMITED.split(line);
399         if (maxCols <= parts.length) {
400           if (trimToNull) {
401             result.put(StringUtils.trimToNull(parts[key]), StringUtils.trimToNull(parts[value]));
402           } else {
403             result.put(parts[key], parts[value]);
404           }
405         }
406       }
407     }
408     return result;
409   }

451   /**
452    * For the given list, finds the index of the lowest value using the given comparator.
453    *
454    * @param values To compare
455    * @param comparator To use
456    * @return The index of the lowest value, or -1 if they are all null
457    */
458   static int lowestValueIndex(List<String> values, Comparator<String> comparator) {
459     int index = 0;
460     String lowestValue = null;
461     for (int i = 0; i < values.size(); i++) {
462       String value = values.get(i);
463       if (lowestValue == null) {
464         lowestValue = value;
465         index = i;
466       } else if (comparator.compare(lowestValue, value) > 0) {
467         lowestValue = value;
468         index = i;
469       }
470     }
471 
472     return lowestValue == null ? -1 : index;
473   }

496   /**
497    * Merges a list of intermediary sort chunk files into a single sorted file. On completion, the intermediary sort
498    * chunk files are deleted.
499    *
500    * @param sortFiles sort chunk files to merge
501    * @param sortedFileWriter writer to merge to. Can already be open and contain data
502    * @param lineComparator To use when determining the order (reuse the one that was used to sort the individual
503    *        files)
504    */
505   public void mergedSortedFiles(List<File> sortFiles, FileWriter sortedFileWriter, Comparator<String> lineComparator)
506     throws IOException {
507     List<BufferedReader> partReaders = new LinkedList<BufferedReader>();
508     try {
509       List<String> partReaderLine = new LinkedList<String>();
510       for (File f : sortFiles) {
511         partReaders.add(new BufferedReader(new FileReader(f)));
512       }
513       boolean moreData = false;
514       // load first line in
515       for (BufferedReader partReader : partReaders) {
516         String partLine = partReader.readLine();
517         if (partLine != null) {
518           moreData = true;
519         }
520         // we still add the "null" to keep the partReaders and partLineReader indexes in sync - ALWAYS
521         partReaderLine.add(partLine);
522       }
523       // keep going until all readers are exhausted
524       while (moreData) {
525         int index = lowestValueIndex(partReaderLine, lineComparator);
526         if (index >= 0) {
527           sortedFileWriter.write(partReaderLine.get(index));
528           sortedFileWriter.write("\n");
529           BufferedReader r = partReaders.get(index);
530           String partLine = r.readLine();
531           // TODO: Synchronization on local variable?
532           synchronized (partReaderLine) {
533             partReaderLine.add(index, partLine);
534             partReaderLine.remove(index + 1);
535           }
536         } else {
537           moreData = false;
538         }
539       }
540     } finally {
541       for (BufferedReader b : partReaders) {
542         try {
543           b.close();
544         } catch (RuntimeException e) {
545         }
546       }
547       // I assume it periodically flushes anyway, so only need to do once at end...
548       sortedFileWriter.flush();
549       sortedFileWriter.close();
550       // delete (intermediary) sort chunk files, only the sorted file remains
551       for (File f : sortFiles) {
552         f.delete();
553       }
554     }
555   }

559    *
560    * This method is not reliable when the sort field may contain Unicode codepoints outside the Basic Multilingual Plane,
561    * i.e. above \uFFFF. In that case, the sort order differs from Java's String sort order.  This should not be a problem
562    * for most usage; the Supplementary Multilingual Planes contain ancient scripts, emojis, arrows and so on.
563    *
564    * @param input To sort
565    * @param sorted The sorted version of the input excluding ignored header lines (see ignoreHeaderLines)
566    * @param column the column that keeps the values to sort on
567    * @param columnDelimiter the delimiter that separates columns in a row
568    * @param enclosedBy optional column enclosing character, e.g. a double quote for CSVs
569    * @param newlineDelimiter the chars used for new lines, usually \n, \n\r or \r
570    * @param ignoreHeaderLines number of beginning lines to ignore, e.g. headers

572   public void sort(File input, File sorted, String encoding, int column, String columnDelimiter, Character enclosedBy,

574     Comparator<String> lineComparator;
575     if (enclosedBy == null) {
576       lineComparator = new LineComparator(column, columnDelimiter);
577     } else {
578       lineComparator = new LineComparator(column, columnDelimiter, enclosedBy);
579     }
580     sort(input, sorted, encoding, column, columnDelimiter, enclosedBy, newlineDelimiter, ignoreHeaderLines,
581       lineComparator, false);
582   }

585   /**
586    * Sorts the input file into the output file using the supplied delimited line parameters.
587    *
588    * This method is not reliable when the sort field may contain Unicode codepoints outside the Basic Multilingual Plane,
589    * i.e. above \uFFFF. In that case, the sort order differs from Java's String sort order.  This should not be a problem
590    * for most usage; the Supplementary Multilingual Planes contain ancient scripts, emojis, arrows and so on.
591    *
592    * @param input To sort
593    * @param sorted The sorted version of the input excluding ignored header lines (see ignoreHeaderLines)
594    * @param column the column that keeps the values to sort on
595    * @param columnDelimiter the delimiter that separates columns in a row
596    * @param enclosedBy optional column enclosing character, e.g. a double quote for CSVs
597    * @param newlineDelimiter the chars used for new lines, usually \n, \r\n or \r
598    * @param ignoreHeaderLines number of beginning lines to ignore, e.g. headers
599    * @param lineComparator used to sort the output
600    * @param ignoreCase ignore case order, this parameter couldn't have any effect if the LineComparator is used
601    */
602   public void sort(File input, File sorted, String encoding, int column, String columnDelimiter, Character enclosedBy,
603     String newlineDelimiter, int ignoreHeaderLines, Comparator<String> lineComparator, boolean ignoreCase)
604     throws IOException {
605     LOG.debug("sorting " + input.getAbsolutePath() + " as new file " + sorted.getAbsolutePath());
606     if (encoding == null) {
607       LOG.warn("No encoding specified, assume UTF-8");
608       encoding = FileUtils.UTF8;
609     }
610     // if the id is in the first column, first try sorting via shell as its the fastest we can get
611     if (!sortInGnu(input, sorted, encoding, ignoreHeaderLines, column, columnDelimiter, newlineDelimiter, ignoreCase)) {
612       LOG.debug("No GNU sort available, using native Java sorting");
613       sortInJava(input, sorted, encoding, lineComparator, ignoreHeaderLines);
614     }
615   }

617   /**
618    * Sorts the input file into the output file using the supplied lineComparator.
619    *
620    * @param input To sort
621    * @param sorted The sorted version of the input excluding ignored header lines (see ignoreHeaderLines)
622    * @param lineComparator To use during comparison
623    * @param ignoreHeaderLines number of beginning lines to ignore, e.g. headers
624    */
625   public void sortInJava(File input, File sorted, String encoding, Comparator<String> lineComparator,
626     int ignoreHeaderLines) throws IOException {
627     LOG.debug("Sorting File[" + input.getAbsolutePath() + ']');
628     long start = System.currentTimeMillis();
629     List<File> sortFiles = new LinkedList<File>();
630     BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(input), encoding));
631     List<String> headerLines = new LinkedList<String>();
632     try {
633       String line = br.readLine();
634       int fileCount = 0;
635 
636       List<String> linesToSort = new LinkedList<String>();
637       while (line != null) {
638         if (ignoreHeaderLines > 0) {
639           headerLines.add(line);
640           ignoreHeaderLines--;
641         } else {
642           linesToSort.add(line);
643 
644           // if buffer is full, then sort and write to file
645           if (linesToSort.size() == linesPerMemorySort) {
646             sortFiles.add(sortAndWrite(input, encoding, lineComparator, fileCount, linesToSort));
647             linesToSort = new LinkedList<String>();
648             fileCount++;
649           }
650         }
651         line = br.readLine();
652       }
653       // catch the last lot
654       if (!linesToSort.isEmpty()) {
655         sortFiles.add(sortAndWrite(input, encoding, lineComparator, fileCount, linesToSort));
656       }
657     } finally {
658       br.close();
659     }
660     LOG.debug(
661       sortFiles.size() + " sorted file chunks created in " + (System.currentTimeMillis() - start) / 1000 + " secs");
662 
663     // now merge the sorted files into one single sorted file
664     FileWriter sortedFileWriter = new FileWriter(sorted);
665     // first write the old header lines if existing
666     for (String h : headerLines) {
667       sortedFileWriter.write(h);
668       sortedFileWriter.write("\n");
669     }
670     mergedSortedFiles(sortFiles, sortedFileWriter, lineComparator);
671 
672     LOG.debug(
673       "File " + input.getAbsolutePath() + " sorted successfully using " + sortFiles.size() + " parts to do sorting in "
674         + (System.currentTimeMillis() - start) / 1000 + " secs");
675   }

678   /**
679    * Splits the supplied file into files of set line size and with a suffix.
680    *
681    * @param input To split up
682    * @param linesPerOutput Lines per split file
683    * @param extension The file extension to use - e.g. ".txt"
684    * @return The split files
685    */
686   public List<File> split(File input, int linesPerOutput, String extension) throws IOException {
687     LOG.debug("Splitting File[" + input.getAbsolutePath() + ']');
688     long timer = System.currentTimeMillis();
689     List<File> splitFiles = new LinkedList<File>();
690     BufferedReader br = new BufferedReader(new FileReader(input));
691     String line = br.readLine();
692     int fileCount = 0;
693     File splitFile = getChunkFile(input, fileCount);
694     fileCount++;
695     splitFiles.add(splitFile);
696     FileWriter fw = new FileWriter(splitFile);
697     try {
698       int lineCount = 0;
699       while (line != null) {
700         if (lineCount == linesPerOutput) {
701           fw.flush();
702           fw.close();
703           splitFile = getChunkFile(input, fileCount);
704           splitFiles.add(splitFile);
705           // is ok to reuse, as last one is closed, and this will always get closed - see finally below
706           fw = new FileWriter(splitFile);
707           fileCount++;
708           lineCount = 0;
709         }
710         fw.write(line);
711         fw.write("\n");
712         line = br.readLine();
713         lineCount++;
714       }
715       fw.flush();
716     } finally {
717       fw.close();
718     }
719     LOG.debug("File[" + input.getAbsolutePath() + "] split successfully into[" + splitFiles.size() + "] parts in secs["
720       + (1 + System.currentTimeMillis() - timer) / 1000 + "]");
721     return splitFiles;
722   }

724   /**
725    * Test whether we have a new enough version of GNU Sort that supports (primarily) the -k option with a start and end
726    * column.
727    *
728    * Mac OS only includes an old version of GNU sort, and will fail this test.
729    */
730   private boolean gnuSortAvailable() {
731     if (gnuSortAvailable != null) {
732       return gnuSortAvailable;
733     }
734 
735     try {
736       String command = "sort -k1,1 -t',' --ignore-case /dev/null";
737       LOG.debug("Testing capability of GNU sort with command: {}", command);
738 
739       Process process = new ProcessBuilder("/bin/sh", "-c", command).start();
740       int exitValue = process.waitFor();
741 
742       if (exitValue == 0) {
743         LOG.debug("GNU sort is capable");
744         gnuSortAvailable = true;
745       } else {
746         LOG.warn("GNU sort does not exist or is too old, and will not be used.  Sorting large files will be slow.",
747             new InputStreamUtils().readEntireStream(process.getErrorStream()).replace('\n', ' '));
748         gnuSortAvailable = false;
749       }
750     } catch (Exception e) {
751       LOG.warn("GNU sort does not exist or is too old, and will not be used.  Sorting large files will be slow.", e);
752       gnuSortAvailable = false;
753     }
754 
755     return gnuSortAvailable;
756   }

758   /**
759    * sort a text file via an external GNU sort command:
760    * sorting tabs at 3rd column, numerical reverse order
761    * sort -t$'\t' -k3 -o sorted.txt col2007.txt
762    * <p/>
763    * The GNU sort based sorting is extremely efficient and much, much faster than the current sortInJava method. It is
764    * locale aware though and we only want the native C sorting locale. See
765    * http://www.gnu.org/software/coreutils/faq/coreutils-faq.html#Sort-does-not-sort-in-normal-order_0021
766    * <p/>
767    * Example C sort order:
768    * <p/>
769    * <pre>
770    * 1 oOdontoceti
771    * 10 gGlobicephala melaena melaena Traill
772    * 100 gGlobicephala melaena melaena Traill
773    * 101 gGlobicephala melaena melaena Traill
774    * 11 pPontoporia Gray
775    * 12 pPontoporia blainvillei Gervais and d'Orbigny
776    * 120 iInia d'Orbigny
777    * 121 iInia geoffrensis Blainville
778    * 2 sSusuidae
779    * 20 cCetacea
780    * Amphiptera
781    * Amphiptera pacifica Giglioli
782    * Anarnak Lacpde
783    * Balaena mangidach Chamisso
784    * amphiptera
785    * amphiptera pacifica Giglioli
786    * anarnak Lacpde
787    * balaena mangidach Chamisso
788    * </pre>
789    */
790   protected boolean sortInGnu(File input, File sorted, String encoding, int ignoreHeaderLines, int column,
791     String columnDelimiter, String lineDelimiter, boolean ignoreCase) throws IOException {
792     String command;
793     // GNU sort is checked for use when:
794     //  line delimiter is \n
795     //  column delimiter is not \n or '
796     //. sort version is sufficient to include start and end column (-k 1,1).
797     // Use the --debug option to sort if working on this code.
798     if (lineDelimiter == null || !lineDelimiter.contains("\n") ||
799         (columnDelimiter != null && (columnDelimiter.contains("\n") || columnDelimiter.contains("'"))) ||
800         !gnuSortAvailable()) {
801       LOG.debug("Cannot use GNU sort on this file");
802       return false;
803     }
804 
805     // keep header rows
806     boolean success = false;
807     try {
808       LinkedList<String> cmds = new LinkedList<String>();
809       cmds.add("/bin/sh");
810       cmds.add("-c");
811       cmds.add("");
812       ProcessBuilder pb = new ProcessBuilder(cmds);
813       Map<String, String> env = pb.environment();
814       env.clear();
815       // make sure we use the C locale for sorting
816       env.put("LC_ALL", "C");
817 
818       String sortArgs = String.format(" %s -k%d,%d -t'%s'",
819         ignoreCase ? "--ignore-case" : "", column+1, column+1, columnDelimiter);
820 
821       if (ignoreHeaderLines > 0) {
822         // copy header lines
823         command = "head -n " + ignoreHeaderLines + ' ' + input.getAbsolutePath() + " > " + sorted.getAbsolutePath();
824         LOG.debug("Issue external command: {}", command);
825         cmds.removeLast();
826         cmds.add(command);
827         Process process = pb.start();
828         int exitValue = process.waitFor();
829         if (exitValue != 0) {
830           LOG.warn("Error sorting file (copying header lines) with GNU head");
831           return false;
832         }
833 
834         // do the sorting ignoring the header rows
835         command = "sed " + ignoreHeaderLines + "d " + input.getAbsolutePath() + " | "
836             + "sort " + sortArgs
837             + " >> " + sorted.getAbsolutePath();
838       } else {
839         // do sorting directly, we don't have header rows
840         command = "sort " + sortArgs + " -o " + sorted.getAbsolutePath() + ' ' + input.getAbsolutePath();
841       }
842 
843       LOG.debug("Issue external command: {}", command);
844       cmds.removeLast();
845       cmds.add(command);
846       Process process = pb.start();
847       // get the stdout and stderr from the command that was run
848       InputStream err = process.getErrorStream();
849       int exitValue = process.waitFor();
850       if (exitValue == 0) {
851         LOG.debug("Successfully sorted file with GNU sort");
852         success = true;
853       } else {
854         LOG.warn("Error sorting file with GNU sort");
855         InputStreamUtils isu = new InputStreamUtils();
856         System.err.append(isu.readEntireStream(err));
857       }
858     } catch (Exception e) {
859       LOG.warn("Caught Exception using GNU sort", e);
860     }
861     return success;
862   }

864   /**
865    * Sorts the lines and writes to file using the
866    *
867    * @param input File to base the name on
868    * @param lineComparator To compare the lines for sorting
869    * @param fileCount Used for the file name
870    * @param linesToSort To actually sort
871    * @return The written file
872    */
873   private File sortAndWrite(File input, String encoding, Comparator<String> lineComparator, int fileCount,
874     List<String> linesToSort) throws IOException {
875     long start = System.currentTimeMillis();
876     Collections.sort(linesToSort, lineComparator);
877     // When implementing a comparator, make it SUPER quick!!!
878     LOG.debug(
879       "Collections.sort took msec[" + (System.currentTimeMillis() - start) + "] to sort records[" + linesToSort.size()
880         + ']');
881     File sortFile = getChunkFile(input, fileCount);
882     Writer fw = new OutputStreamWriter(new FileOutputStream(sortFile), encoding);
883     try {
884       for (String s : linesToSort) {
885         fw.write(s);
886         fw.write("\n");
887       }
888     } finally {
889       fw.close();
890     }
891     return sortFile;
892   }
