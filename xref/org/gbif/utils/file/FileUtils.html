98    /**
99     * Reads a file and returns a unique set of multiple columns from lines which are no comments (starting with #) and
100    * trims whitespace.
101    *
102    * @param source the UTF-8 encoded text file with tab delimited columns
103    * @param resultSet the set implementation to be used. Will not be cleared before reading!
104    * @param column variable length argument of column indices to process
105    * @return set of column rows
106    */
107   public static Set<String> columnsToSet(InputStream source, Set<String> resultSet, int... column)
108       throws IOException {
109     LineIterator lines = getLineIterator(source);
110     int maxCols = 0;
111     for (int c : column) {
112       if (c > maxCols) {
113         maxCols = c;
114       }
115     }
116     while (lines.hasNext()) {
117       String line = lines.nextLine().trim();
118       // ignore comments
119       if (!ignore(line)) {
120         String[] parts = TAB_DELIMITED.split(line);
121         if (maxCols <= parts.length) {
122           for (int c : column) {
123             String cell = parts[c].trim();
124             resultSet.add(cell);
125           }
126         }
127       }
128     }
129     return resultSet;
130   }

132   public static void copyStreams(InputStream in, OutputStream out) throws IOException {
133     // write the file to the file specified
134     int bytesRead;
135     byte[] buffer = new byte[8192];
136 
137     while ((bytesRead = in.read(buffer, 0, 8192)) != -1) {
138       out.write(buffer, 0, bytesRead);
139     }
140 
141     out.close();
142     in.close();
143   }

249   /**
250    * Converts the byte size into human-readable format.
251    * Support both SI and byte format.
252    */
253   public static String humanReadableByteCount(long bytes, boolean si) {
254     int unit = si ? 1000 : 1024;
255     if (bytes < unit) {
256       return bytes + " B";
257     }
258     int exp = (int) (Math.log(bytes) / Math.log(unit));
259     String pre = (si ? "kMGTPE" : "KMGTPE").charAt(exp - 1) + (si ? "" : "i");
260     return String.format("%.1f %sB", bytes / Math.pow(unit, exp), pre);
261   }

285   public static ByteBuffer readByteBuffer(File file, int bufferSize) throws IOException {
286     ByteBuffer bbuf = ByteBuffer.allocate(bufferSize);
287     BufferedInputStream f = new BufferedInputStream(new FileInputStream(file), bufferSize);
288 
289     int b;
290     while ((b = f.read()) != -1) {
291       if (!bbuf.hasRemaining()) {
292         break;
293       }
294       bbuf.put((byte) b);
295     }
296     f.close();
297 
298     return bbuf;
299   }

399   /**
400    * Read a hashmap from a tab delimited utf8 file, ignoring commented rows starting with #.
401    *
402    * @param source tab delimited input stream to read
403    * @param key column number to use as key
404    * @param value column number to use as value
405    * @param trimToNull if true trims map entries to null
406    */
407   public static Map<String, String> streamToMap(
408       InputStream source, Map<String, String> result, int key, int value, boolean trimToNull)
409       throws IOException {
410     LineIterator lines = getLineIterator(source);
411     int maxCols = key > value ? key : value + 1;
412     while (lines.hasNext()) {
413       String line = lines.nextLine();
414       // ignore comments
415       if (!ignore(line)) {
416         String[] parts = TAB_DELIMITED.split(line);
417         if (maxCols <= parts.length) {
418           if (trimToNull) {
419             result.put(StringUtils.trimToNull(parts[key]), StringUtils.trimToNull(parts[value]));
420           } else {
421             result.put(parts[key], parts[value]);
422           }
423         }
424       }
425     }
426     return result;
427   }

472   /**
473    * For the given list, finds the index of the lowest value using the given comparator.
474    *
475    * @param values To compare
476    * @param comparator To use
477    * @return The index of the lowest value, or -1 if they are all null
478    */
479   static int lowestValueIndex(List<String> values, Comparator<String> comparator) {
480     int index = 0;
481     String lowestValue = null;
482     for (int i = 0; i < values.size(); i++) {
483       String value = values.get(i);
484       if (lowestValue == null) {
485         lowestValue = value;
486         index = i;
487       } else if (comparator.compare(lowestValue, value) > 0) {
488         lowestValue = value;
489         index = i;
490       }
491     }
492 
493     return lowestValue == null ? -1 : index;
494   }

521   /**
522    * Merges a list of intermediary sort chunk files into a single sorted file. On completion, the intermediary sort
523    * chunk files are deleted.
524    *
525    * @param sortFiles sort chunk files to merge
526    * @param sortedFileWriter writer to merge to. Can already be open and contain data
527    * @param lineComparator To use when determining the order (reuse the one that was used to sort the individual
528    *        files)
529    */
530   public void mergedSortedFiles(
531       List<File> sortFiles, OutputStreamWriter sortedFileWriter, Comparator<String> lineComparator)
532       throws IOException {
533     List<BufferedReader> partReaders = new LinkedList<>();
534     try {
535       List<String> partReaderLine = new LinkedList<>();
536       for (File f : sortFiles) {
537         // Use UTF-8 sort order.
538         partReaders.add(
539             new BufferedReader(
540                 new InputStreamReader(new FileInputStream(f), StandardCharsets.UTF_8)));
541       }
542       boolean moreData = false;
543       // load first line in
544       for (BufferedReader partReader : partReaders) {
545         String partLine = partReader.readLine();
546         if (partLine != null) {
547           moreData = true;
548         }
549         // we still add the "null" to keep the partReaders and partLineReader indexes in sync -
550         // ALWAYS
551         partReaderLine.add(partLine);
552       }
553       // keep going until all readers are exhausted
554       while (moreData) {
555         int index = lowestValueIndex(partReaderLine, lineComparator);
556         if (index >= 0) {
557           sortedFileWriter.write(partReaderLine.get(index));
558           sortedFileWriter.write("\n");
559           BufferedReader r = partReaders.get(index);
560           String partLine = r.readLine();
561           // TODO: Synchronization on local variable?
562           synchronized (partReaderLine) {
563             partReaderLine.add(index, partLine);
564             partReaderLine.remove(index + 1);
565           }
566         } else {
567           moreData = false;
568         }
569       }
570     } finally {
571       for (BufferedReader b : partReaders) {
572         try {
573           b.close();
574         } catch (RuntimeException e) {
575         }
576       }
577       // I assume it periodically flushes anyway, so only need to do once at end...
578       sortedFileWriter.flush();
579       sortedFileWriter.close();
580       // delete (intermediary) sort chunk files, only the sorted file remains
581       for (File f : sortFiles) {
582         f.delete();
583       }
584     }
585   }

588    * Sorts the input file into the output file using the supplied delimited line parameters.
589    *
590    * This method is not reliable when the sort field may contain Unicode codepoints outside the Basic Multilingual Plane,
591    * i.e. above \uFFFF. In that case, the sort order differs from Java's String sort order.  This should not be a problem
592    * for most usage; the Supplementary Multilingual Planes contain ancient scripts, emojis, arrows and so on.
593    *
594    * @param input To sort
595    * @param sorted The sorted version of the input excluding ignored header lines (see ignoreHeaderLines)
596    * @param column the column that keeps the values to sort on
597    * @param columnDelimiter the delimiter that separates columns in a row
598    * @param enclosedBy optional column enclosing character, e.g. a double quote for CSVs
599    * @param newlineDelimiter the chars used for new lines, usually \n, \n\r or \r
600    * @param ignoreHeaderLines number of beginning lines to ignore, e.g. headers
601    */
602   public void sort(
603       File input,
604       File sorted,
605       String encoding,
606       int column,
607       String columnDelimiter,
608       Character enclosedBy,
609       String newlineDelimiter,
610       int ignoreHeaderLines)
611       throws IOException {
612     Comparator<String> lineComparator;
613     if (enclosedBy == null) {
614       lineComparator = new LineComparator(column, columnDelimiter);
615     } else {
616       lineComparator = new LineComparator(column, columnDelimiter, enclosedBy);
617     }
618     sort(
619         input,
620         sorted,
621         encoding,
622         column,
623         columnDelimiter,
624         enclosedBy,
625         newlineDelimiter,
626         ignoreHeaderLines,
627         lineComparator,
628         false);

631   /**
632    * Sorts the input file into the output file using the supplied delimited line parameters.
633    *
634    * This method is not reliable when the sort field may contain Unicode codepoints outside the Basic Multilingual Plane,
635    * i.e. above \uFFFF. In that case, the sort order differs from Java's String sort order.  This should not be a problem
636    * for most usage; the Supplementary Multilingual Planes contain ancient scripts, emojis, arrows and so on.
637    *
638    * TODO: This method is globally synchronized, in case multiple sorts are attempted to the same file simultaneously.
639    * This could be improved to allow synchronizing against the destination file, rather than for all sorts.
640    *
641    * @param input To sort
642    * @param sorted The sorted version of the input excluding ignored header lines (see ignoreHeaderLines)
643    * @param column the column that keeps the values to sort on
644    * @param columnDelimiter the delimiter that separates columns in a row
645    * @param enclosedBy optional column enclosing character, e.g. a double quote for CSVs
646    * @param newlineDelimiter the chars used for new lines, usually \n, \r\n or \r
647    * @param ignoreHeaderLines number of beginning lines to ignore, e.g. headers
648    * @param lineComparator used to sort the output
649    * @param ignoreCase ignore case order, this parameter couldn't have any effect if the LineComparator is used
650    */
651   public void sort(
652       File input,
653       File sorted,
654       String encoding,
655       int column,
656       String columnDelimiter,
657       Character enclosedBy,
658       String newlineDelimiter,
659       int ignoreHeaderLines,
660       Comparator<String> lineComparator,
661       boolean ignoreCase)
662       throws IOException {
663     LOG.debug("Sorting " + input.getAbsolutePath() + " as new file " + sorted.getAbsolutePath());
664     if (encoding == null) {
665       LOG.warn("No encoding specified, assume UTF-8");
666       encoding = FileUtils.UTF8;
667     }
668     synchronized (sortLock) {
669       if (sorted.exists()) {
670         // Delete a file, which will allow processes with it open to continue reading it.
671         // The GNU sort truncates and appends, which would mean a partial read otherwise.
672         LOG.warn("Deleting existed sorted file {}", sorted.getAbsoluteFile());
673         sorted.delete();
674       }
675       // if the id is in the first column, first try sorting via shell as its the fastest we can get
676       if (!sortInGnu(
677           input,
678           sorted,
679           encoding,
680           ignoreHeaderLines,
681           column,
682           columnDelimiter,
683           newlineDelimiter,
684           ignoreCase)) {
685         LOG.debug("No GNU sort available, using native Java sorting");
686         sortInJava(input, sorted, encoding, lineComparator, ignoreHeaderLines);
687       }
688     }
689   }

704       int ignoreHeaderLines)
705       throws IOException {
706     LOG.debug("Sorting File[" + input.getAbsolutePath() + ']');
707     long start = System.currentTimeMillis();
708     List<File> sortFiles = new LinkedList<>();
709     BufferedReader br =
710         new BufferedReader(new InputStreamReader(new FileInputStream(input), encoding));
711     List<String> headerLines = new LinkedList<>();
712     try {
713       String line = br.readLine();
714       int fileCount = 0;
715 
716       List<String> linesToSort = new LinkedList<>();
717       while (line != null) {
718         if (ignoreHeaderLines > 0) {
719           headerLines.add(line);
720           ignoreHeaderLines--;
721         } else {
722           linesToSort.add(line);
723 
724           // if buffer is full, then sort and write to file
725           if (linesToSort.size() == linesPerMemorySort) {
726             sortFiles.add(sortAndWrite(input, encoding, lineComparator, fileCount, linesToSort));
727             linesToSort = new LinkedList<>();
728             fileCount++;
729           }
730         }
731         line = br.readLine();
732       }
733       // catch the last lot
734       if (!linesToSort.isEmpty()) {
735         sortFiles.add(sortAndWrite(input, encoding, lineComparator, fileCount, linesToSort));
736       }
737     } finally {
738       br.close();
739     }
740     LOG.debug(
741         sortFiles.size()
742             + " sorted file chunks created in "
743             + (System.currentTimeMillis() - start) / 1000
744             + " secs");
745 
746     // now merge the sorted files into one single sorted file
747     FileWriter sortedFileWriter = new FileWriter(sorted);
748     // first write the old header lines if existing
749     for (String h : headerLines) {
750       sortedFileWriter.write(h);
751       sortedFileWriter.write("\n");
752     }
753     mergedSortedFiles(sortFiles, sortedFileWriter, lineComparator);
754 
755     LOG.debug(
756         "File "
757             + input.getAbsolutePath()
758             + " sorted successfully using "
759             + sortFiles.size()
760             + " parts to do sorting in "
761             + (System.currentTimeMillis() - start) / 1000
762             + " secs");
763   }

765   /**
766    * Splits the supplied file into files of set line size and with a suffix.
767    *
768    * @param input To split up
769    * @param linesPerOutput Lines per split file
770    * @param extension The file extension to use - e.g. ".txt"
771    * @return The split files
772    */
773   public List<File> split(File input, int linesPerOutput, String extension) throws IOException {
774     LOG.debug("Splitting File[" + input.getAbsolutePath() + ']');
775     long timer = System.currentTimeMillis();
776     List<File> splitFiles = new LinkedList<>();
777     // Use ISO-8859-1 as a binary-safe encoding.
778     BufferedReader br =
779         new BufferedReader(
780             new InputStreamReader(new FileInputStream(input), StandardCharsets.ISO_8859_1));
781     String line = br.readLine();
782     int fileCount = 0;
783     File splitFile = getChunkFile(input, fileCount);
784     fileCount++;
785     splitFiles.add(splitFile);
786     OutputStreamWriter fw =
787         new OutputStreamWriter(new FileOutputStream(splitFile), StandardCharsets.ISO_8859_1);
788     try {
789       int lineCount = 0;
790       while (line != null) {
791         if (lineCount == linesPerOutput) {
792           fw.flush();
793           fw.close();
794           splitFile = getChunkFile(input, fileCount);
795           splitFiles.add(splitFile);
796           // is ok to reuse, as last one is closed, and this will always get closed - see finally
797           // below
798           fw = new OutputStreamWriter(new FileOutputStream(splitFile), StandardCharsets.ISO_8859_1);
799           fileCount++;
800           lineCount = 0;
801         }
802         fw.write(line);
803         fw.write("\n");
804         line = br.readLine();
805         lineCount++;
806       }
807       fw.flush();
808     } finally {
809       fw.close();
810     }
811     LOG.debug(
812         "File["
813             + input.getAbsolutePath()
814             + "] split successfully into["
815             + splitFiles.size()
816             + "] parts in secs["
817             + (1 + System.currentTimeMillis() - timer) / 1000
818             + "]");
819     return splitFiles;
820   }

861    * sorting tabs at 3rd column, numerical reverse order
862    * sort -t$'\t' -k3 -o sorted.txt col2007.txt
863    * <p/>
864    * The GNU sort based sorting is extremely efficient and much, much faster than the current sortInJava method. It is
865    * locale aware though and we only want the native C sorting locale. See
866    * http://www.gnu.org/software/coreutils/faq/coreutils-faq.html#Sort-does-not-sort-in-normal-order_0021
867    * <p/>
868    * Example C sort order:
869    * <p/>
870    * <pre>
871    * 1 oOdontoceti

874    * 101 gGlobicephala melaena melaena Traill
875    * 11 pPontoporia Gray
876    * 12 pPontoporia blainvillei Gervais and d'Orbigny
877    * 120 iInia d'Orbigny
878    * 121 iInia geoffrensis Blainville
879    * 2 sSusuidae
880    * 20 cCetacea
881    * Amphiptera
882    * Amphiptera pacifica Giglioli
883    * Anarnak Lacépède
884    * Balaena mangidach Chamisso
885    * amphiptera
886    * amphiptera pacifica Giglioli
887    * anarnak Lacépède
888    * balaena mangidach Chamisso
889    * </pre>
890    */
891   protected boolean sortInGnu(
892       File input,
893       File sorted,
894       String encoding,
895       int ignoreHeaderLines,
896       int column,
897       String columnDelimiter,
898       String lineDelimiter,
899       boolean ignoreCase)
900       throws IOException {
901     String command;
902     // GNU sort is checked for use when:
903     // • line delimiter is \n
904     // • column delimiter is set and we're not using the first column
905     // • sort version is sufficient to include start and end column (-k 1,1).
906     // Use the --debug option to sort if working on this code.
907     if (lineDelimiter == null
908         || !lineDelimiter.contains("\n")
909         || (columnDelimiter != null && column > 0)
910         || !gnuSortAvailable()) {
911       LOG.debug("Cannot use GNU sort on this file");
912       return false;
913     }
914 
915     // keep header rows
916     boolean success = false;
917     try {
918       LinkedList<String> cmds = new LinkedList<>();
919       cmds.add("/bin/sh");
920       cmds.add("-c");
921       cmds.add("");
922       ProcessBuilder pb = new ProcessBuilder(cmds);
923       Map<String, String> env = pb.environment();
924 
925       // clear the environment, but keep specified temp working directory
926       env.keySet().removeIf(key -> !(key.equals("TMPDIR")));
927       if (System.getProperty("java.io.tmpdir") != null) {
928         env.put("TMPDIR", System.getProperty("java.io.tmpdir"));
929       }
930       // make sure we use the C locale for sorting
931       env.put("LC_ALL", "C");
932 
933       String sortArgs =
934           String.format(
935               " %s -k%d,%d -t'%s'",
936               ignoreCase ? "--ignore-case" : "", column + 1, column + 1, columnDelimiter);
937 
938       if (ignoreHeaderLines > 0) {
939         // copy header lines
940         command =
941             "head -n "
942                 + ignoreHeaderLines
943                 + ' '
944                 + input.getAbsolutePath()
945                 + " > "
946                 + sorted.getAbsolutePath();
947         LOG.debug("Issue external command: {}", command);
948         cmds.removeLast();
949         cmds.add(command);
950         Process process = pb.start();
951         int exitValue = process.waitFor();
952         if (exitValue != 0) {
953           LOG.warn("Error sorting file (copying header lines) with GNU head");
954           return false;
955         }
956 
957         // do the sorting ignoring the header rows
958         command =
959             "sed "
960                 + ignoreHeaderLines
961                 + "d "
962                 + input.getAbsolutePath()
963                 + " | "
964                 + "sort "
965                 + sortArgs
966                 + " >> "
967                 + sorted.getAbsolutePath();
968       } else {
969         // do sorting directly, we don't have header rows
970         command =
971             "sort " + sortArgs + " -o " + sorted.getAbsolutePath() + ' ' + input.getAbsolutePath();
972       }
973 
974       LOG.debug("Issue external command: {}", command);
975       cmds.removeLast();
976       cmds.add(command);
977       Process process = pb.start();
978       // get the stdout and stderr from the command that was run
979       InputStream err = process.getErrorStream();
980       int exitValue = process.waitFor();
981       if (exitValue == 0) {
982         LOG.debug("Successfully sorted file with GNU sort");
983         success = true;
984       } else {
985         LOG.warn("Error sorting file with GNU sort");
986         InputStreamUtilss.html#InputStreamUtils">InputStreamUtils isu = new InputStreamUtils();
987         System.err.append(isu.readEntireStream(err));
988       }
989     } catch (Exception e) {
990       LOG.warn("Caught Exception using GNU sort", e);
991     }
992     return success;
993   }
