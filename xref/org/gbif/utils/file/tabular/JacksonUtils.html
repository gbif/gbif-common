12    /**
13     * Build the default {@link CsvSchema}.
14     * @param delimiterChar
15     * @param endOfLineSymbols
16     * @param quoteChar
17     * @return
18     */
19    static CsvSchema buildCsvSchema(char delimiterChar, String endOfLineSymbols, Character quoteChar) {
20      CsvSchema schema = CsvSchema.emptySchema();
21      schema = schema
22              .withColumnSeparator(delimiterChar)
23              .withLineSeparator(endOfLineSymbols);
24  
25      //quote character is optional
26      schema = quoteChar == null ? schema.withoutQuoteChar() : schema.withQuoteChar(quoteChar);
27      schema = schema.withEscapeChar('\\');
28  
29      return schema;
30    }
