34  import com.fasterxml.jackson.dataformat.csv.CsvMapper;

41  /**
42   * Utility class to rewrite a tabular file (e.g. CSV) into a normalized format.
43   * The regular use case is to allow external tools to work as expected (e.g. unix split, unix sort).
44   */
45  public class TabularFileNormalizer {
46  
47    // A character is considered to be an ISO control character if its code is in
48    // the range '\u0000' through '\u001F' or in the range '\u007F' through '\u009F'.
49    private static CharMatcher CONTROL_CHAR_MATCHER = CharMatcher.JAVA_ISO_CONTROL;
50  
51    public static String NORMALIZED_END_OF_LINE = "\n";
52  
53    /**
54     * Normalizes the provided tabular "file" (provided as {@link Reader} to let the caller deal with charset).
55     * Normalization includes: striping of Control Characters (see {@link #CONTROL_CHAR_MATCHER}),
56     * usage of \n as end-line-character, ensuring there is an end-of-line character on the last line and
57     * removing empty (completely empty) lines.
58     * The normalized content will have unnecessary quotes removed.
59     *
60     * @param source           {@link Path} representing the source
61     * @param destination      {@link Path} representing the destination. If the file already exists it will be overwritten.
62     * @param sourceCharset    optionally, the {@link Charset} of the source. If null UTF-8 will be used.
63     * @param delimiterChar
64     * @param endOfLineSymbols
65     * @param quoteChar        optional
66     *
67     * @return number of lines written
68     *
69     * @throws IOException
70     */
71    public static int normalizeFile(Path source, Path destination, Charset sourceCharset,
72                                    char delimiterChar, String endOfLineSymbols, Character quoteChar) throws IOException {
73      Objects.requireNonNull(source, "source path shall be provided");
74      Objects.requireNonNull(destination, "normalizedWriter shall be provided");
75  
76      int numberOfLine = 0;
77      CsvMapper mapper = new CsvMapper();
78      mapper.enable(CsvParser.Feature.WRAP_AS_ARRAY);
79  
80      mapper.configure(CsvGenerator.Feature.STRICT_CHECK_FOR_QUOTING, true);
81  
82      CsvSchema readerSchema = buildCsvSchema(delimiterChar, endOfLineSymbols, quoteChar);
83      CsvSchema writerSchema = buildCsvSchema(delimiterChar, NORMALIZED_END_OF_LINE, quoteChar);
84  
85      Charset charset = Optional.ofNullable(sourceCharset).orElse(StandardCharsets.UTF_8);
86      try (Reader sourceReader = Files.newBufferedReader(source, charset);
87           Writer writer = Files.newBufferedWriter(destination, charset);
88           MappingIterator<List<String>> it = mapper.readerFor(List.class)
89                   .with(readerSchema)
90                   .readValues(sourceReader);
91           SequenceWriter csvWriter = mapper.writerFor(List.class).with(writerSchema).writeValues(writer)) {
92        Optional<List<String>> line;
93        while (it.hasNext()) {
94          line = normalizeLine(it.next());
95          if (line.isPresent()) {
96            csvWriter.write(line.get());
97            numberOfLine++;
98          }
99        }
100     }
101     catch (IOException ioEx) {
102       //avoid keeping incomplete file
103       Files.deleteIfExists(destination);
104       throw ioEx;
105     }
106     return numberOfLine;
107   }
108 
109   /**
110    * For a given line in a tabular file, normalize it if it contains something.
111    *
112    * @param line
113    *
114    * @return normalized line as String or {@code Optional.empty()} is the line was empty
115    */
116   private static Optional<List<String>> normalizeLine(List<String> line) {
117     if (line == null || line.isEmpty()) {
118       return Optional.empty();
119     }
120 
121     return Optional.of(
122             line.stream()
123                     .map(s -> s == null ? "" : s)
124                     .map(CONTROL_CHAR_MATCHER::removeFrom)
125                     .collect(Collectors.toList()));
126   }
127 }
