55  /**
56   * Utility class to extract metadata {@link TabularFileMetadata} from a tabular file.
57   */
58  public class TabularFileMetadataExtractor {
59  
60    private static final Logger LOG = LoggerFactory.getLogger(TabularFileMetadataExtractor.class);
61    private static final int MAX_SAMPLE_SIZE = 15;
62  
63    // This needs to be large enough to stumble upon a non-ASCII character.
64    private static final int CHARSET_DETECTION_BUFFER_LENGTH = 1024*1024;
65  
66    private TabularFileMetadataExtractor() {
67    }
68  
69    private static final Character[] POTENTIAL_DELIMITER_CHAR = {',', '\t', ';', '|'};
70    private static final Character[] POTENTIAL_QUOTES_CHAR = {'"', '\''};
71  
72    private static final Predicate<LineDelimiterStats> CONTAINS_FREQUENCY = lineStats -> lineStats.getFrequency() > 0;
73    private static final Comparator<Map.Entry<Character, Long>> BY_VALUE_LONG_DESC = Comparator.comparing(Map.Entry::getValue, reverseOrder());
74    private static final BiFunction<Character, Character, Pattern> COMPILE_QUOTE_PATTERN_FCT = (delimiter, quoteChar)
75            -> Pattern.compile("[" + delimiter + "][ ]*[" + quoteChar + "][ ]*[^" + delimiter + "]");

77    /**
78     * Extract metadata from a tabular file using a sample (defined by {@link #MAX_SAMPLE_SIZE}) of the file.
79     * The extraction process is based on the frequency of character in the sample using 3 different approaches.
80     * The method will not return any default value if no delimiter and/or quote character can be found in the sample.
81     * The caller should decide which default values should be used to read the file.
82     *
83     * @param filePath a {@link Path} pointing to a file (not a folder).
84     * @return new {@link TabularFileMetadata}, never null (but the content can be null).
85     * @throws IOException
86     * @throws UnknownCharsetException
87     */
88    public static TabularFileMetadata extractTabularFileMetadata(Path filePath) throws IOException, UnknownCharsetException {
89      Objects.requireNonNull(filePath, "filePath shall be provided");
90      Preconditions.checkArgument(!Files.isDirectory(filePath), "filePath should point to a file, not a directory");
91  
92      Charset encoding;
93      try {
94        encoding = CharsetDetection.detectEncoding(filePath.toFile(), CHARSET_DETECTION_BUFFER_LENGTH);
95        if (encoding == null) {
96          throw new UnknownCharsetException("Unable to detect the file's character encoding");
97        }
98      } catch (IOException e) {
99        throw new UnknownCharsetException(e);
100     }
101 
102     // open a first stream to read a sample of the file
103     List<String> lines = new ArrayList<>();
104     try (BufferedReader bf = Files.newBufferedReader(filePath, encoding)) {
105       String line;
106       do {
107         line = bf.readLine();
108         if(line != null) {
109           lines.add(line);
110         }
111       }
112       while(line != null && lines.size() < MAX_SAMPLE_SIZE);
113     }
114     TabularFileMetadata tabularFileMetadata = extractTabularMetadata(lines);
115     tabularFileMetadata.setEncoding(encoding);
116     return tabularFileMetadata;
117   }

119   /**
120    * Tries to extract the {@link TabularFileMetadata} from a sample of lines of a tabular file.
121    *
122    * @param sample
123    * @return new {@link TabularFileMetadata}, never null (but the content can be null).
124    */
125   static TabularFileMetadata extractTabularMetadata(final List<String> sample) {
126     Objects.requireNonNull(sample, "sample shall be provided");
127     TabularFileMetadata tabularFileMetadata = new TabularFileMetadata();
128 
129     Optional<Character> delimiterFound = getDelimiterChar(sample);
130     final Character delimiter = delimiterFound.orElse(null);
131     if(delimiter == null) {
132       return tabularFileMetadata;
133     }
134 
135     Optional<Character> quoteFound = getHighestCountOf(sample, line -> getQuoteCharWithHighestCount(line, delimiter));
136     final Character quote = quoteFound.orElse(null);
137 
138     tabularFileMetadata.setDelimiter(delimiter);
139     tabularFileMetadata.setQuotedBy(quote);
140 
141     return tabularFileMetadata;
142   }

144   /**
145    * Extract a character from a line using the given function.
146    * Return the character with the highest counts.
147    *
148    * @param sample
149    * @param characterExtractor function to apply on each line to extract a character
150    *
151    * @return
152    */
153   private static Optional<Character> getHighestCountOf(final List<String> sample, final Function<String,
154           Optional<Character>> characterExtractor) {
155 
156     return sample.stream()
157             .map(characterExtractor)
158             .flatMap(o -> o.map(Stream::of).orElseGet(Stream::empty)) //remove Optional wrapper and ignore Optional.empty
159             .collect(Collectors.groupingBy(Function.identity(), counting()))
160             .entrySet()
161             .stream()
162             .sorted(BY_VALUE_LONG_DESC)
163             .findFirst()
164             .map(Map.Entry::getKey);
165   }

167   /**
168    * Given a sample of line, this method tries to determine the delimiter char used.
169    *
170    * @param sample
171    *
172    * @return the determined delimiter or Optional.empty if it can not be determined.
173    */
174   public static Optional<Character> getDelimiterChar(final List<String> sample) {
175 
176     // count the frequency of all possible delimiter for each lines
177     List<LineDelimiterStats> linesStats =
178             computeLineDelimiterStats(sample);
179 
180     // get the distinct set of frequency for each delimiters to check the "stability"
181     Map<Character, Set<Integer>> delimiterDistinctFrequency =
182             computeDelimiterDistinctFrequency(linesStats)
183                     .entrySet().stream()
184                     // filter out delimiter that we never saw
185                     .filter(entry -> entry.getValue().size() > 1 || !entry.getValue().contains(Integer.valueOf(0)))
186                     .sorted(Comparator.comparing(e -> e.getValue().size()))
187                     .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue,
188                             (e1, e2) -> e2, LinkedHashMap::new));
189 
190     // we can have more than one
191     Set<Character> mostStableDelimiter =
192             getAllEqualsToFirst(delimiterDistinctFrequency, (s1,s2) -> s1.size() == s2.size());
193 
194     // get the most used delimiter to check the "overall usage"
195     Map<Character, Integer> delimiterFrequencySums = computeDelimiterFrequencySums(linesStats)
196             .entrySet().stream()
197             .sorted( Comparator.comparing( (Map.Entry<Character, Integer> e) -> e.getValue()).reversed())
198             .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue,
199                     (e1, e2) -> e2, LinkedHashMap::new));
200 
201     Set<Character> mostFrequentDelimiter = getAllEqualsToFirst(delimiterFrequencySums, Integer::equals);
202 
203     //get the highest frequency per line to check for "usage per line"
204     Map<Character, Long> delimiterHighestFrequencyPerLine = computeDelimiterHighestFrequencyPerLine(sample)
205             .entrySet().stream()
206             .sorted( Comparator.comparing( (Map.Entry<Character, Long> e) -> e.getValue()).reversed())
207             .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue,
208                     (e1, e2) -> e2, LinkedHashMap::new));
209     Set<Character> mostFrequentDelimiterPerLine = getAllEqualsToFirst(delimiterHighestFrequencyPerLine,
210            Long::equals);
211 
212     //summary
213     if (LOG.isDebugEnabled()) {
214       LOG.debug("delimiterDistinctFrequency -> " + delimiterDistinctFrequency);
215       LOG.debug("mostStableDelimiter -> " + mostStableDelimiter);
216       LOG.debug("delimiterFrequencySums -> " + delimiterFrequencySums);
217       LOG.debug("mostFrequentDelimiter -> " + mostFrequentDelimiter);
218       LOG.debug("delimiterHighestFrequencyPerLine->" + delimiterHighestFrequencyPerLine);
219       LOG.debug("mostFrequentDelimiterPerLine ->" + mostFrequentDelimiterPerLine);
220     }
221 
222     //if the most stable is also the one that is used to most within the sample
223     Optional<Character> resultCharacter = intersectSingle(mostStableDelimiter, mostFrequentDelimiter);
224     if(resultCharacter.isPresent()) {
225       return resultCharacter;
226     }
227 
228     //otherwise, if the most stable is also the most used based on lines
229     resultCharacter = intersectSingle(mostStableDelimiter, mostFrequentDelimiterPerLine);
230     if(resultCharacter.isPresent()) {
231       return resultCharacter;
232     }
233 
234     //as last resort if the most frequent delimiter overall and by line is the same
235     resultCharacter = intersectSingle(mostFrequentDelimiter, mostFrequentDelimiterPerLine);
236     if(resultCharacter.isPresent()) {
237       return resultCharacter;
238     }
239 
240     // other idea
241     // give more weight to the first line (it is probably the header line where normally no quotes are used)
242     // give more weight to the first "column" (it is probably the "id" line where normally no quotes are used)
243     return Optional.empty();
244   }

246   /**
247    * Return the {@link Character} represents the intersection between 2 sets only if the resulting set represents
248    * a single element.
249    * @param set1
250    * @param set2
251    * @return
252    */
253   private static Optional<Character> intersectSingle(Set<Character> set1, Set<Character> set2) {
254     Sets.SetView<Character> intersection = Sets.intersection(set1, set2);
255     return intersection.size() == 1 ? intersection.stream().findFirst() : Optional.empty();
256   }

258   /**
259    * Given a {@link Map}, return all elements that are equals to the first element (including itself)
260    * based on the provided equals function.
261    * @param map
262    * @param equalsPredicate
263    * @param <T>
264    * @return all elements that are equals to the first one or an empty set if the map is empty
265    */
266   private static <T> Set<Character> getAllEqualsToFirst(Map<Character, T> map, BiFunction<T,T, Boolean> equalsPredicate) {
267 
268     Optional<Map.Entry<Character, T>> firstMapEntry = map.entrySet().stream().findFirst();
269     if(!firstMapEntry.isPresent()){
270       return Collections.EMPTY_SET;
271     }
272 
273     final T firstValue = firstMapEntry.get().getValue();
274     return map.entrySet().stream().filter( e -> equalsPredicate.apply(firstValue,e.getValue()))
275             .map(Map.Entry::getKey).collect(Collectors.toSet());
276   }

278   /**
279    * For each element(line) of the sample, compute a {@link LineDelimiterStats} for each delimiter.
280    * Note: delimiter that are not used within a line will be included with the frequency 0.
281    * @param sample
282    * @return new List, never null
283    */
284   static List<LineDelimiterStats> computeLineDelimiterStats(List<String> sample){
285     return sample.stream()
286             .map(TabularFileMetadataExtractor::lineToLineDelimiterStats)
287             .flatMap(List::stream)
288             .collect(Collectors.toList());
289   }

291   /**
292    * Compute the stats for each potential delimiters on a line.
293    * @param line
294    * @return
295    */
296   private static List<LineDelimiterStats> lineToLineDelimiterStats(String line) {
297     return Arrays.stream(POTENTIAL_DELIMITER_CHAR)
298             .map( delimiter -> new LineDelimiterStats(delimiter, StringUtils.countMatches(line, delimiter)))
299             .collect(Collectors.toList());
300   }

302   /**
303    * For each {@link LineDelimiterStats}, collect the distinct frequency (count) of each delimiter.
304    * This gives us an idea of the "stability" of each delimiter across the sample.
305    * Note that since quotes are not handled, noise can be introduced if a quoted cells use the delimiter.
306    *
307    * See unit test for examples of when this method will be affected by noise.
308    *
309    * The most stable delimiter is normally defined by the {@link Character} returned by the methods where
310    * the list of distinct frequency is the smallest in size excluding cases where the list contains only the element
311    * representing 0 as Integer (which means the delimiter was never used).
312    *
313    * @param linesStats
314    *
315    * @return
316    */
317   static Map<Character, Set<Integer>> computeDelimiterDistinctFrequency(List<LineDelimiterStats> linesStats) {
318     return linesStats.stream()
319             .collect(
320                     Collectors.groupingBy(LineDelimiterStats::getDelimiter,
321                             Collectors.mapping(LineDelimiterStats::getFrequency, toSet())));
322   }

324   /**
325    * For each line, check the delimiter that is used the most.
326    * Return the count of each delimiter.
327    * @param lines
328    * @return
329    */
330   static Map<Character, Long> computeDelimiterHighestFrequencyPerLine(List<String> lines) {
331     return lines.stream()
332             .map(TabularFileMetadataExtractor::getDelimiterWithHighestCount2)
333             .flatMap(o -> o.map(Stream::of).orElseGet(Stream::empty)) //remove Optional wrapper and ignore Optional.empty
334             .collect(Collectors.groupingBy(LineDelimiterStats::getDelimiter, counting()));
335   }
336 
337   /**
338    * For {@link LineDelimiterStats}, sum the frequency (count) of each delimiter.
339    * This gives us an idea of the overall usage of each delimiter across the sample.
340    * Note that since quotes are not handled, noise can be introduced if a quoted cell uses the delimiter.
341    *
342    * See unit test for examples of when this method will be affected by noise.
343    *
344    * @param linesStats
345    *
346    * @return
347    */
348   static Map<Character, Integer> computeDelimiterFrequencySums(List<LineDelimiterStats> linesStats) {
349     return linesStats.stream()
350             .filter(CONTAINS_FREQUENCY)
351             .collect(
352                     Collectors.groupingBy(LineDelimiterStats::getDelimiter,
353                             Collectors.summingInt(LineDelimiterStats::getFrequency)));
354   }

356   /**
357    * Given a line, get the delimiter with the highest count if any can be found.
358    * Note: quotes are ignored in the count so a delimiter used inside quotes will be counted.
359    *
360    * @param line line of text to analyse
361    *
362    * @return
363    */
364   static Optional<Character> getDelimiterWithHighestCount(String line) {
365     int highestCount = 0;
366     Character highestCountDelimiter = null;
367     for (Character delimiter : POTENTIAL_DELIMITER_CHAR) {
368       int currentCount = StringUtils.countMatches(line, delimiter);
369       if (currentCount > highestCount) {
370         highestCount = currentCount;
371         highestCountDelimiter = delimiter;
372       }
373     }
374     return Optional.ofNullable(highestCountDelimiter);
375   }
376 
377   static Optional<LineDelimiterStats> getDelimiterWithHighestCount2(String line) {
378     int highestCount = 0;
379    // Character highestCountDelimiter = null;
380     LineDelimiterStats lineDelimiterStats = null;
381     for (Character delimiter : POTENTIAL_DELIMITER_CHAR) {
382       int currentCount = StringUtils.countMatches(line, delimiter);
383       if (currentCount > highestCount) {
384         highestCount = currentCount;
385         lineDelimiterStats = new LineDelimiterStats(delimiter,highestCount);
386       }
387     }
388     return Optional.ofNullable(lineDelimiterStats);
389   }

391   /**
392    * Given a line and a delimiter, try to determine the quoting character if any can be found.
393    * To check if a quote character is used we run a regex to check for a delimiter followed by a quoting character.
394    *
395    * @param line line of text to analyse
396    * @param delimiter delimiter used in the line of text
397    *
398    * @return
399    */
400   static Optional<Character> getQuoteCharWithHighestCount(String line, Character delimiter) {
401     int highestCount = 0;
402     Character highestCountQuoteChar = null;
403     for (Character quoteChar : POTENTIAL_QUOTES_CHAR) {
404       int currentCount = 0;
405       Matcher m = COMPILE_QUOTE_PATTERN_FCT.apply(delimiter, quoteChar).matcher(line);
406       while (m.find()) {
407         currentCount++;
408       }
409       if (currentCount > highestCount) {
410         highestCount = currentCount;
411         highestCountQuoteChar = quoteChar;
412       }
413     }
414     return Optional.ofNullable(highestCountQuoteChar);
415   }

417   /**
418    * Inner representation of stats (frequency) of a delimiter
419    */
420   static class LineDelimiterStats {
421     private Character delimiter;
422     private int frequency;
423     LineDelimiterStats(Character delimiter, int frequency) {
424       this.delimiter = delimiter;
425       this.frequency = frequency;
426     }
427 
428     Character getDelimiter() {
429       return delimiter;
430     }
431 
432     int getFrequency() {
433       return frequency;
434     }
435   }
