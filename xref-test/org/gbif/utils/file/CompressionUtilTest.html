38    public static File createTempDirectory() throws IOException {
39  
40      final File temp = File.createTempFile("temp", Long.toString(System.nanoTime()));
41  
42      if (!temp.delete()) {
43        throw new IOException("Could not delete temp file: " + temp.getAbsolutePath());
44      }
45  
46      if (!temp.mkdir()) {
47        throw new IOException("Could not create temp directory: " + temp.getAbsolutePath());
48      }
49      temp.deleteOnExit();
50  
51      return temp;
52    }

54    private static void assureEqualContent(List<File> result, String metaContent, String dataContent) throws IOException {
55      for (File f : result) {
56        String x = FileUtils.readFileToString(f, "utf-8");
57        if ("meta.xml".equals(f.getName())) {
58          assertEquals(metaContent, x);
59        } else if ("quote_in_quote.csv".equals(f.getName())) {
60          assertEquals(dataContent, x);
61        } else {
62          fail("unexpected file");
63        }
64      }
65    }

67    public File classpathFile(String path) {
68      File f = null;
69      // relative path. Use classpath instead
70      URL url = getClass().getClassLoader().getResource(path);
71      if (url != null) {
72        f = new File(url.getFile());
73      }
74      return f;
75    }

77    @Test
78    public void testDecompress() throws IOException {
79      // meta.xml
80      File meta = classpathFile("compression/archive/meta.xml");
81      String metaContent = FileUtils.readFileToString(meta, "utf-8");
82      // quote_in_quote.csv
83      File data = classpathFile("compression/archive/quote_in_quote.csv");
84      String dataContent = FileUtils.readFileToString(data, "utf-8");
85  
86      File tmpDir = createTempDirectory();
87      File testArchiveFile = classpathFile("compression/archive.zip");
88      List<File> result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
89      assertEquals(2, result.size());
90      assureEqualContent(result, metaContent, dataContent);
91  
92      FileUtils.cleanDirectory(tmpDir);
93      testArchiveFile = classpathFile("compression/archive.tgz");
94      result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
95      assertEquals(2, result.size());
96      assureEqualContent(result, metaContent, dataContent);
97  
98      FileUtils.cleanDirectory(tmpDir);
99      testArchiveFile = classpathFile("compression/archive-zip.dat");
100     result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
101     assertEquals(2, result.size());
102     assureEqualContent(result, metaContent, dataContent);
103 
104     FileUtils.cleanDirectory(tmpDir);
105     testArchiveFile = classpathFile("compression/archive-tgz.dat");
106     result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
107     assertEquals(2, result.size());
108     assureEqualContent(result, metaContent, dataContent);
109 
110     FileUtils.cleanDirectory(tmpDir);
111     testArchiveFile = classpathFile("compression/cate.zip");
112     result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
113     assertEquals(3, result.size());
114   }

139   public void testUnzipFolderDoNotKeepSubdirectoriesOrHiddenFiles() throws IOException {
140     File tmpDir = createTempDirectory();
141     File testZippedFolder = classpathFile("compression/withSubdirsAndHiddenFiles.zip");
142     List<File> files = CompressionUtil.unzipFile(tmpDir, testZippedFolder, false);
143     assertEquals(9, files.size()); // 9 files, 0 directories
144     assertTrue(new File(tmpDir, "dwca.zip").exists());
145     assertTrue(new File(tmpDir, "eml.xml").exists());
146     assertTrue(new File(tmpDir, "publication.log").exists());
147     assertTrue(new File(tmpDir, "resource.xml").exists());
148     assertTrue(new File(tmpDir, "test4.rtf").exists());
149     assertTrue(new File(tmpDir, "taxon.log").exists());
150     assertTrue(new File(tmpDir, "taxon.txt").exists());
151     assertTrue(new File(tmpDir, "taxonshort.log").exists());
152     assertTrue(new File(tmpDir, "taxonshort.txt").exists());
153     // assert subdirectory is removed
154     File sourceDir = new File(tmpDir, "sources");
155     assertFalse(sourceDir.exists());
156     // assert wrapping root directory is removed
157     File rootDir = new File(tmpDir, "withSubdirsAndHiddenFiles");
158     assertFalse(rootDir.exists());
159     // assert hidden files are removed
160     assertFalse(new File(tmpDir, ".hidden1").exists());
161     assertFalse(new File(tmpDir, "/sources/.hidden2").exists());
162     // assert .DS_Store removed
163     assertFalse(new File(tmpDir, ".DS_Store").exists());
164     // assert __MACOSX removed
165     assertFalse(new File(tmpDir, "__MACOSX").exists());
166   }

236     assertTrue(new File(dwcaDir, "meta.xml").exists());
237     assertTrue(new File(dwcaDir, "eml.xml").exists());
238 
239     // assert hidden files and directories are removed
240     assertFalse(new File(tmpDir, ".svn").exists());
241     assertFalse(new File(tmpDir, "/sources/.svn").exists());

260   @Test
261   public void testZipFolder() throws IOException {
262     File zipWithDirs = File.createTempFile("aha", ".zip");
263     System.out.println(zipWithDirs.getAbsolutePath());
264 
265     File zipWithoutDirs = File.createTempFile("aha", ".zip");
266     System.out.println(zipWithoutDirs.getAbsolutePath());
267     //tmp.deleteOnExit();
268     File testFolder = classpathFile("charsets");
269     // remember how many files we have in the root folder, exlcuding files in subdirectories
270     final int rootFileNum = testFolder.listFiles().length;
271 
272     CompressionUtil.zipDir(testFolder, zipWithDirs, true);
273     CompressionUtil.zipDir(testFolder, zipWithoutDirs, false);
274 
275     assertNotEquals(zipWithDirs.length(), zipWithoutDirs.length());
276 
277     // now decompress the subdir zip and make sure we get the same amount of root files
278     File tmpDir = org.gbif.utils.file.FileUtils.createTempDir();
279     tmpDir.deleteOnExit();
280     CompressionUtil.unzipFile(tmpDir, zipWithDirs, true);
281     int decompressedRootFileNum = tmpDir.listFiles().length;
282     assertEquals(rootFileNum, decompressedRootFileNum);
283   }
