76     * tests deleting directory recursively.
77     */
78      public
79      void testDeleteRecursive() throws IOException {
80      File topFile = File.createTempFile("top", ".tmp");
81      File middleFile = File.createTempFile("middle", ".tmp", topFile.getParentFile());
82      File downFile = File.createTempFile("down", ".tmp", middleFile.getParentFile());
83  
84      assertTrue(topFile.getParentFile().exists());
85      assertTrue(topFile.exists());
86      assertTrue(middleFile.getParentFile().exists());
87      assertTrue(middleFile.exists());
88      assertTrue(downFile.getParentFile().exists());
89      assertTrue(downFile.exists());
90  
91      FileUtils.deleteDirectoryRecursively(topFile.getParentFile());
92  
93      assertFalse(topFile.getParentFile().exists());
94      assertFalse(topFile.exists());
95      assertFalse(middleFile.getParentFile().exists());
96      assertFalse(middleFile.exists());
97      assertFalse(downFile.getParentFile().exists());
98      assertFalse(downFile.exists());
99    }

132   /**
133    * Sorting strings containing characters which are surrogate pairs, meaning Unicode characters beyond U+FFFF, will
134    * give different results between GNU Sort and a Java String comparator.
135    *
136    * "ﬂ LATIN SMALL LIGATURE FL" is U+FB02.
137    * "ð LINEAR B IDEOGRAM B241 CHARIOT" is U+100CD.
138    *
139    * GNU sort will use this order, based on the value of the whole character.
140    *
141    * Java represents ð as a surrogate pair \ud800\udccd in UTF-16, and sorts based on parts of pairs. Therefore, it
142    * gives the wrong order.
143    */
144   @Ignore("Expected to fail")
145   @Test
146   public void testSortingUnicodeFile() throws IOException {
147     FileUtils futils = new FileUtils();
148     final int IDCOLUMN = 0;
149 
150     File source = FileUtils.getClasspathFile("sorting/unicode-supplementary-multilingual-plane.txt");
151     File gnuSorted = File.createTempFile("gbif-common-file-sort", "sorted-gnu.txt");
152     File javaSorted = File.createTempFile("gbif-common-file-sort", "sorted-java.txt");
153     gnuSorted.deleteOnExit();
154     javaSorted.deleteOnExit();
155 
156     futils.sort(source, gnuSorted, ENCODING, IDCOLUMN, "\t", null, "\n", 0);
157     // The columnDelimiter of ' prevents GNU Sort from being used.
158     futils.sort(source, javaSorted, ENCODING, IDCOLUMN, "'", null, "\n", 0);
159 
160     // read file
161     BufferedReader gnuBr = new BufferedReader(new InputStreamReader(new FileInputStream(gnuSorted), "UTF-8"));
162     BufferedReader javaBr = new BufferedReader(new InputStreamReader(new FileInputStream(javaSorted), "UTF-8"));
163 
164     int line = 0;
165     String gnuRow, javaRow;
166     while ((gnuRow = gnuBr.readLine()) != null) {
167       javaRow = javaBr.readLine();
168 
169       line++;
170 
171       System.out.println(gnuRow + "\t\t\t\t" + javaRow);
172 
173       assertEquals("Line "+line, gnuRow, javaRow);
174     }
175   }

220   /**
221    * Tests sorting by a column with uneven length strings as the sort column.
222    *
223    * The order musn't be different depending whether the column is last or not.
224    *
225    * The "-k×,×" argument to sort is essential here, otherwise the delimiter from the following column is part of the sort order.
226    */
227   @Test
228   public void testSortingUnevenLengths() throws IOException {
229     FileUtils futils = new FileUtils();
230 
231     File source = FileUtils.getClasspathFile("sorting/uneven_lengths_col1.txt");
232     File sorted = File.createTempFile("sort-test", "uneven_lengths_col1.txt");
233     sorted.deleteOnExit();
234     futils.sort(source, sorted, ENCODING, 0, ";", null, "\n", 0);
235 
236     List<String> sortedStrings = FileUtils.streamToList(new FileInputStream(sorted), "UTF-8");
237     assertEquals("980-sp10;x" , sortedStrings.get(0));
238     assertEquals("980-sp100;x", sortedStrings.get(1));
239     assertEquals("980-sp101;x", sortedStrings.get(2));
240 
241     File source2 = FileUtils.getClasspathFile("sorting/uneven_lengths_col2.txt");
242     File sorted2 = File.createTempFile("sort-test", "uneven_lengths_col2.txt");
243     sorted.deleteOnExit();
244     futils.sort(source2, sorted2, ENCODING, 1, ";", null, "\n", 0);
245 
246     List<String> sortedStrings2 = FileUtils.streamToList(new FileInputStream(sorted2), "UTF-8");
247     assertEquals("x;980-sp10" , sortedStrings2.get(0));
248     assertEquals("x;980-sp100", sortedStrings2.get(1));
249     assertEquals("x;980-sp101", sortedStrings2.get(2));
250   }

320   /**
321    * Test that ensures the chunk file is deleted at the end of sortInJava method. Otherwise, unwanted chunk files
322    * will be left over.
323    */
324   @Test
325   public void testSortInJava() throws IOException {
326     File source = FileUtils.getClasspathFile("sorting/taxon.txt");
327     File sorted = File.createTempFile("gbif-common-file-sort", "taxon_sorted.txt");
328     sorted.deleteOnExit();
329     FileUtils futils = new FileUtils();
330     Comparator<String> lineComparator = new LineComparator(0, "\t");
331     futils.sortInJava(source, sorted, ENCODING, lineComparator, 1);
332 
333     // the chunk file should NOT exist
334     File chunkFile = new File(source.getParent(), "taxon_0txt");
335     assertFalse(chunkFile.exists());
336 
337     // the sorted file should exist
338     System.out.println(sorted.getAbsolutePath());
339     assertTrue(sorted.exists());
340 
341     // read file
342     BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(sorted), "UTF-8"));
343     int line = 0;
344     while (true) {
345       line++;
346       String row = br.readLine();
347       if (row == null) {
348         break;
349       }
350       // first line (smallest ID)
351       if (line == 2) {
352         assertTrue(row.startsWith("118701359"));
353       }
354       // last line (largest ID)
355       else if (line == 8) {
356         assertTrue(row.startsWith("120320038"));
357       }
358     }
