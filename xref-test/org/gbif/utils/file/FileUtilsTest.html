95    public void testDeleteRecursive() throws IOException {
96      File topFile = File.createTempFile("top", ".tmp");
97      File middleFile = File.createTempFile("middle", ".tmp", topFile.getParentFile());
98      File downFile = File.createTempFile("down", ".tmp", middleFile.getParentFile());
99  
100     assertTrue(topFile.getParentFile().exists());
101     assertTrue(topFile.exists());
102     assertTrue(middleFile.getParentFile().exists());
103     assertTrue(middleFile.exists());
104     assertTrue(downFile.getParentFile().exists());
105     assertTrue(downFile.exists());
106 
107     FileUtils.deleteDirectoryRecursively(topFile.getParentFile());
108 
109     assertFalse(topFile.getParentFile().exists());
110     assertFalse(topFile.exists());
111     assertFalse(middleFile.getParentFile().exists());
112     assertFalse(middleFile.exists());
113     assertFalse(downFile.getParentFile().exists());
114     assertFalse(downFile.exists());
115   }

150   /**
151    * Sorting strings containing characters which are surrogate pairs, meaning Unicode characters beyond U+FFFF, will
152    * give different results between GNU Sort and a Java String comparator.
153    *
154    * "ﬂ LATIN SMALL LIGATURE FL" is U+FB02.
155    * "ð LINEAR B IDEOGRAM B241 CHARIOT" is U+100CD.
156    *
157    * GNU sort will use this order, based on the value of the whole character.
158    *
159    * Java represents ð as a surrogate pair \ud800\udccd in UTF-16, and sorts based on parts of pairs. Therefore, it
160    * gives the wrong order.
161    */
162   @Disabled("Expected to fail")
163   @Test
164   public void testSortingUnicodeFile() throws IOException {
165     FileUtils futils = new FileUtils();
166     final int IDCOLUMN = 0;
167 
168     File source =
169         FileUtils.getClasspathFile("sorting/unicode-supplementary-multilingual-plane.txt");
170     File gnuSorted = File.createTempFile("gbif-common-file-sort", "sorted-gnu.txt");
171     File javaSorted = File.createTempFile("gbif-common-file-sort", "sorted-java.txt");
172     gnuSorted.deleteOnExit();
173     javaSorted.deleteOnExit();
174 
175     futils.sort(source, gnuSorted, ENCODING, IDCOLUMN, "\t", null, "\n", 0);
176     // The columnDelimiter of ' prevents GNU Sort from being used.
177     futils.sort(source, javaSorted, ENCODING, IDCOLUMN, "'", null, "\n", 0);
178 
179     // read file
180     BufferedReader gnuBr =
181         new BufferedReader(
182             new InputStreamReader(new FileInputStream(gnuSorted), StandardCharsets.UTF_8));
183     BufferedReader javaBr =
184         new BufferedReader(
185             new InputStreamReader(new FileInputStream(javaSorted), StandardCharsets.UTF_8));
186 
187     int line = 0;
188     String gnuRow, javaRow;
189     while ((gnuRow = gnuBr.readLine()) != null) {
190       javaRow = javaBr.readLine();
191 
192       line++;
193 
194       System.out.println(gnuRow + "\t\t\t\t" + javaRow);
195 
196       assertEquals("Line " + line, gnuRow, javaRow);
197     }
198   }

387   /**
388    * Test that ensures the chunk file is deleted at the end of sortInJava method. Otherwise, unwanted chunk files
389    * will be left over.
390    */
391   @Test
392   public void testSortInJava() throws IOException {
393     File source = FileUtils.getClasspathFile("sorting/taxon.txt");
394     File sorted = File.createTempFile("gbif-common-file-sort", "taxon_sorted.txt");
395     sorted.deleteOnExit();
396     FileUtils futils = new FileUtils();
397     Comparator<String> lineComparator = new LineComparator(0, "\t");
398     futils.sortInJava(source, sorted, ENCODING, lineComparator, 1);
399 
400     // the chunk file should NOT exist
401     File chunkFile = new File(source.getParent(), "taxon_0txt");
402     assertFalse(chunkFile.exists());
403 
404     // the sorted file should exist
405     System.out.println(sorted.getAbsolutePath());
406     assertTrue(sorted.exists());
407 
408     // read file
409     BufferedReader br =
410         new BufferedReader(
411             new InputStreamReader(new FileInputStream(sorted), StandardCharsets.UTF_8));
412     int line = 0;
413     while (true) {
414       line++;
415       String row = br.readLine();
416       if (row == null) {
417         break;
418       }
419       // first line (smallest ID)
420       if (line == 2) {
421         assertTrue(row.startsWith("118701359"));
422       }
423       // last line (largest ID)
424       else if (line == 8) {
425         assertTrue(row.startsWith("120320038"));
426       }
427     }
