82     * tests deleting directory recursively.

84    @Test
85    public void testDeleteRecursive() throws IOException {
86      File topDirectory = Files.createTempDirectory("top").toFile();
87      File middleDirectory = new File(topDirectory, "middle");
88      middleDirectory.mkdir();
89      File bottomDirectory = new File(middleDirectory, "bottom");
90      bottomDirectory.mkdir();
91      File bottomFile = new File(bottomDirectory, "bottom");
92      FileUtils.touch(bottomFile);
93  
94      assertTrue(topDirectory.getParentFile().exists());
95      assertTrue(topDirectory.exists());
96      assertTrue(middleDirectory.getParentFile().exists());
97      assertTrue(middleDirectory.exists());
98      assertTrue(bottomDirectory.getParentFile().exists());
99      assertTrue(bottomDirectory.exists());
100     assertTrue(bottomFile.exists());
101 
102     FileUtils.deleteDirectoryRecursively(topDirectory);
103 
104     assertTrue(topDirectory.getParentFile().exists());
105     assertFalse(topDirectory.exists());
106     assertFalse(middleDirectory.getParentFile().exists());
107     assertFalse(middleDirectory.exists());
108     assertFalse(bottomDirectory.getParentFile().exists());
109     assertFalse(bottomDirectory.exists());
110     assertFalse(bottomFile.exists());
111   }

146   /**
147    * Sorting strings containing characters which are surrogate pairs, meaning Unicode characters beyond U+FFFF, will
148    * give different results between GNU Sort and a Java String comparator.
149    *
150    * "ﬂ LATIN SMALL LIGATURE FL" is U+FB02.
151    * "ð LINEAR B IDEOGRAM B241 CHARIOT" is U+100CD.
152    *
153    * GNU sort will use this order, based on the value of the whole character.
154    *
155    * Java represents ð as a surrogate pair \ud800\udccd in UTF-16, and sorts based on parts of pairs. Therefore, it
156    * gives the wrong order.
157    */
158   @Disabled("Expected to fail")
159   @Test
160   public void testSortingUnicodeFile() throws IOException {
161     FileUtils futils = new FileUtils();
162     final int IDCOLUMN = 0;
163 
164     File source =
165         FileUtils.getClasspathFile("sorting/unicode-supplementary-multilingual-plane.txt");
166     File gnuSorted = File.createTempFile("gbif-common-file-sort", "sorted-gnu.txt");
167     File javaSorted = File.createTempFile("gbif-common-file-sort", "sorted-java.txt");
168     gnuSorted.deleteOnExit();
169     javaSorted.deleteOnExit();
170 
171     futils.sort(source, gnuSorted, ENCODING, IDCOLUMN, "\t", null, "\n", 0);
172     // The columnDelimiter of ' prevents GNU Sort from being used.
173     futils.sort(source, javaSorted, ENCODING, IDCOLUMN, "'", null, "\n", 0);
174 
175     // read file
176     BufferedReader gnuBr =
177         new BufferedReader(
178             new InputStreamReader(new FileInputStream(gnuSorted), StandardCharsets.UTF_8));
179     BufferedReader javaBr =
180         new BufferedReader(
181             new InputStreamReader(new FileInputStream(javaSorted), StandardCharsets.UTF_8));
182 
183     int line = 0;
184     String gnuRow, javaRow;
185     while ((gnuRow = gnuBr.readLine()) != null) {
186       javaRow = javaBr.readLine();
187 
188       line++;
189 
190       System.out.println(gnuRow + "\t\t\t\t" + javaRow);
191 
192       assertEquals("Line " + line, gnuRow, javaRow);
193     }
194   }

383   /**
384    * Test that ensures the chunk file is deleted at the end of sortInJava method. Otherwise, unwanted chunk files
385    * will be left over.
386    */
387   @Test
388   public void testSortInJava() throws IOException {
389     File source = FileUtils.getClasspathFile("sorting/taxon.txt");
390     File sorted = File.createTempFile("gbif-common-file-sort", "taxon_sorted.txt");
391     sorted.deleteOnExit();
392     FileUtils futils = new FileUtils();
393     Comparator<String> lineComparator = new LineComparator(0, "\t");
394     futils.sortInJava(source, sorted, ENCODING, lineComparator, 3);
395 
396     // the chunk file should NOT exist
397     File chunkFile = new File(source.getParent(), "taxon_0txt");
398     assertFalse(chunkFile.exists());
399 
400     // the sorted file should exist
401     System.out.println(sorted.getAbsolutePath());
402     assertTrue(sorted.exists());
403 
404     // read file
405     BufferedReader br =
406         new BufferedReader(
407             new InputStreamReader(new FileInputStream(sorted), StandardCharsets.UTF_8));
408     int line = 0;
409     while (true) {
410       line++;
411       String row = br.readLine();
412       if (row == null) {
413         break;
414       }
415       // first line (smallest ID)
416       if (line == 4) {
417         assertTrue(row.startsWith("118701359"));
418       }
419       // last line (largest ID)
420       else if (line == 10) {
421         assertTrue(row.startsWith("120320038"));
422       }
423     }
