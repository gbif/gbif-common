16  import java.io.InputStream;
17  import javax.xml.parsers.SAXParser;
18  import javax.xml.parsers.SAXParserFactory;
19  
20  import org.apache.commons.io.IOUtils;
21  import org.apache.commons.io.input.BOMInputStream;
22  import org.junit.Test;
23  import org.xml.sax.ext.DefaultHandler2;
24  
25  import static org.junit.Assert.assertEquals;
26  
27  /**
28   * @author markus
29   */
30  public class BomSafeInputStreamWrapperTest {
31  
32  
33    static SAXParserFactory SAX_FACTORY = SAXParserFactory.newInstance();
34  
35    static {
36      SAX_FACTORY.setNamespaceAware(true);
37      SAX_FACTORY.setValidating(false);
38    }
39  
40    /**
41     * The Java SAX Parser is known to have problems with UTF8 file that contain a proper BOM markup:
42     * http://bugs.java.com/bugdatabase/view_bug.do?bug_id=4508058
43     * https://de.wikipedia.org/wiki/Byte_Order_Mark
44     *
45     *  Make sure the SAX Parser can handle any valid UTF files by using a BomSafeInputStreamWrapper stream.
46     */
47    @Test
48    public void testSaxParser() throws Exception {
49      SAXParser p = SAX_FACTORY.newSAXParser();
50      for (String f : new String[]{"utf8","utf8bom","utf16le","utf16be"}) {
51        String fn = "/sax/" + f + ".xml";
52        System.out.println(fn);
53        InputStream is = getClass().getResourceAsStream(fn);
54        p.parse(is, new DefaultHandler2());
55  
56        is = new BOMInputStream(getClass().getResourceAsStream(fn));
57        p.parse(is, new DefaultHandler2());
58      }
59    }
60  
61    @Test
62    public void testUTF16Stream() throws Exception {
63      // should be the exact same bytes
64  
65      byte[] b1 = IOUtils.toByteArray(getClass().getResourceAsStream("/sax/utf16le.xml"));
66      byte[] b2 = IOUtils.toByteArray(new BOMInputStream(getClass().getResourceAsStream("/sax/utf16le.xml")));
67  
68      assertEquals(b1.length, b2.length);
69      int idx=0;
70      for (byte b : b1) {
71        assertEquals(b, b2[idx++]);
72      }
73    }
74  
75  
76    @Test
77    public void testBomSafeInputStreamWrapper() throws Exception {
78      // test no bom
79      InputStream in = new BomSafeInputStreamWrapper(getClass().getResourceAsStream("/charsets/utf-8_names.txt"));
80      int x = in.read();
81      int y = in.read();
82      int z = in.read();
83      in.close();
84      assertEquals(35, x);
85      assertEquals(35, y);
86      assertEquals(35, z);
87  
88      in = new BomSafeInputStreamWrapper(getClass().getResourceAsStream("/charsets/utf-8_bom_names.txt"));
89      x = in.read();
90      y = in.read();
91      z = in.read();
92      in.close();
93      assertEquals(35, x);
94      assertEquals(35, y);
95      assertEquals(35, z);
96  
97      in = new BomSafeInputStreamWrapper(getClass().getResourceAsStream("/charsets/utf-16LE_names.txt"));
98      x = in.read();
99      y = in.read();
100     z = in.read();
101     in.close();
102     assertEquals(35, x);
103     assertEquals(0, y);
104     assertEquals(35, z);
105 
106     in =
107       new BomSafeInputStreamWrapper(getClass().getResourceAsStream("/charsets/utf-16LE_bom_names.txt"));
108     x = in.read();
109     y = in.read();
110     z = in.read();
111     in.close();
112     assertEquals(35, x);
113     assertEquals(0, y);
114     assertEquals(35, z);
115 
116     in =
117       new BomSafeInputStreamWrapper(getClass().getResourceAsStream("/charsets/utf-16BE_bom_names.txt"));
118     x = in.read();
119     y = in.read();
120     z = in.read();
121     in.close();
122     assertEquals(0, x);
123     assertEquals(35, y);
124     assertEquals(0, z);
125 
126     in =
127       new BomSafeInputStreamWrapper(getClass().getResourceAsStream("/charsets/utf-16BE_bom_names.txt"));
128     x = in.read();
129     y = in.read();
130     z = in.read();
131     in.close();
132     assertEquals(0, x);
133     assertEquals(35, y);
134     assertEquals(0, z);
135   }
