17  package org.gbif.utils.file;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.nio.charset.Charset;
22  
23  import org.apache.commons.lang3.StringUtils;
24  import org.junit.Test;
25  
26  import static org.gbif.utils.file.CharsetDetection.detectEncoding;
27  
28  import static org.junit.Assert.assertEquals;
29  
30  /**
31   * @author markus
32   */
33  public class CharsetDetectionTest {
34  
35    @Test
36    public void testCP1252Encoding() throws IOException {
37      File test = FileUtils.getClasspathFile("charsets/cp1252-test.txt");
38  
39      Charset encoding = detectEncoding(test);
40  
41      assertEquals("windows1252", encoding.displayName().replace("-", "").toLowerCase());
42    }
43  
44  
45    @Test
46    public void testEncodingDetection() throws IOException {
47      String[] files =
48        new String[] {"iso-8859-1_names.txt", "macroman_names.txt", "utf-16BE_bom_names.txt", "utf-16BE_names.txt",
49          "utf-16LE_bom_names.txt", "utf-16LE_names.txt", "utf-8_bom_names.txt", "utf-8_names.txt",
50          "windows1252_names.txt"};
51      for (String fn : files) {
52        File test = FileUtils.getClasspathFile("charsets/" + fn);
53  
54        Charset encoding = detectEncoding(test);
55        String expected = StringUtils.substringBefore(fn, "_");
56  
57        // x-MacRoman is alias for MacRoman used on unix, therefore remove x-
58        assertEquals(expected.replace("-", "").toLowerCase(),
59          encoding.displayName().toLowerCase().replace("x-mac", "mac").replace("-", ""));
60      }
61    }
62  
63    @Test
64    public void testEncodingDetectionKyles() throws IOException {
65      String[] files =
66        new String[] {"utf-8_arabic.csv", "utf-8_japanese.csv", "utf-8_korean.csv", "utf-8_latin.csv", "utf-8_no-bom.csv",
67          "utf-8_traditional-chinese.csv", "utf-8_.csv", "utf-16BE_no-bom.csv", "utf-16BE_.csv",
68          "utf-16LE_little-endian-no-bom.csv", "utf-16LE_little-endian.csv"};
69      for (String fn : files) {
70        File test = FileUtils.getClasspathFile("charsets/kyle/" + fn);
71  
72        Charset encoding = detectEncoding(test);
73        String expected = StringUtils.substringBefore(fn, "_");
74  
75        assertEquals(expected.replace("-", "").toLowerCase(), encoding.displayName().toLowerCase().replace("-", ""));
76      }
77    }
78  }
