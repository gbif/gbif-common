15    @Test
16    public void testMap() {
17      Map<String, Double> map = new HashMap<String, Double>(4);
18      map.put("A", 99.5);
19      map.put("B", 67.4);
20      map.put("C", 67.5);
21      map.put("D", 67.3);
22  
23      Map<String, Double> mapSorted = MapUtils.sortByValue(map);
24  
25      assertEquals(4, mapSorted.size());
26  
27      Queue<String> expected = new LinkedList<String>();
28      expected.add("D");
29      expected.add("B");
30      expected.add("C");
31      expected.add("A");
32      for (String k : mapSorted.keySet()) {
33        assertEquals(expected.poll(), k);
34      }
35    }
