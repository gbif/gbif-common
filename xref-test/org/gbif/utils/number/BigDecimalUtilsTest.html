14  package org.gbif.utils.number;
15  
16  import java.math.BigDecimal;
17  
18  import org.junit.jupiter.api.Test;
19  
20  import static org.junit.jupiter.api.Assertions.assertEquals;
21  import static org.junit.jupiter.api.Assertions.assertNotEquals;
22  import static org.junit.jupiter.api.Assertions.assertThrows;
23  
24  /**
25   *
26   * Tests for BigDecimal utility class
27   *
28   */
29  public class BigDecimalUtilsTest {
30  
31    /**
32     * This test is simply to illustrate why we use the BigDecimal(String) constructor.
33     * new BigDecimal(23.1) equals 23.10000000000000142108547152020037174224853515625
34     */
35    @Test
36    public void testBigDecimalConstructor() {
37      assertNotEquals(new BigDecimal("23.1"), new BigDecimal(23.1d));
38    }
39  
40    @Test
41    public void testBigDecimalConstructorNull() {
42      assertThrows(NullPointerException.class, () -> BigDecimalUtils.fromDouble(null, false));
43    }
44  
45    /**
46     * This test is simply to show why we are not using BigDecimal stripTrailingZeros function.
47     */
48    @Test
49    public void demonstrateBigDecimalStripTrailingZerosEquality() {
50      assertNotEquals(new BigDecimal("500.0").stripTrailingZeros(), new BigDecimal("500"));
51  
52      // but, compareTo function will work as expected
53      assertEquals(0, new BigDecimal("500").compareTo(new BigDecimal("500.0").stripTrailingZeros()));
54    }
55  
56    @Test
57    public void testFromDouble() {
58      assertEquals(new BigDecimal("23.4"), BigDecimalUtils.fromDouble(23.4, true));
59      assertEquals(new BigDecimal("23.4"), BigDecimalUtils.fromDouble(23.4, false));
60  
61      assertEquals(new BigDecimal("23"), BigDecimalUtils.fromDouble(23d, true));
62      assertEquals(new BigDecimal("23.0"), BigDecimalUtils.fromDouble(23d, false));
63  
64      assertEquals(new BigDecimal("23"), BigDecimalUtils.fromDouble(23.0, true));
65      assertEquals(new BigDecimal("23.0"), BigDecimalUtils.fromDouble(23.0, false));
66  
67      assertEquals(new BigDecimal("23"), BigDecimalUtils.fromDouble(23.0000000000, true));
68      assertEquals(new BigDecimal("23.0"), BigDecimalUtils.fromDouble(23.0000000000, false));
69  
70      assertEquals(
71          new BigDecimal("23.000000000001"), BigDecimalUtils.fromDouble(23.000000000001, true));
72      assertEquals(
73          new BigDecimal("23.000000000001"), BigDecimalUtils.fromDouble(23.000000000001, false));
74  
75      assertEquals(new BigDecimal("500"), BigDecimalUtils.fromDouble(500d, true));
76      assertEquals(new BigDecimal("500.0"), BigDecimalUtils.fromDouble(500d, false));
77  
78      assertEquals(new BigDecimal("50000"), BigDecimalUtils.fromDouble(50000d, true));
79      assertEquals(new BigDecimal("50000.0"), BigDecimalUtils.fromDouble(50000d, false));
80  
81      assertEquals(new BigDecimal("50000.01"), BigDecimalUtils.fromDouble(50000.01, true));
82      assertEquals(new BigDecimal("50000.01"), BigDecimalUtils.fromDouble(50000.01, false));
83    }
84  }
