99    @Test
100   public void testThenJoin() {
101     assertEquals("", StringUtils.thenJoin(org.apache.commons.lang3.StringUtils::trimToNull, (String[]) null));
102     assertEquals("", StringUtils.thenJoin(org.apache.commons.lang3.StringUtils::trimToNull, (String) null));
103     assertEquals("", StringUtils.thenJoin(org.apache.commons.lang3.StringUtils::trimToNull, "", " "));
104     assertEquals("x", StringUtils.thenJoin(org.apache.commons.lang3.StringUtils::trimToNull, "", "  x "));
105     assertEquals("x y", StringUtils.thenJoin(org.apache.commons.lang3.StringUtils::trimToNull, "x", "  y "));
106     assertEquals("x   y ", StringUtils.thenJoin(Function.identity(), "x", "  y "));
107   }
